--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList ;
	if LuaDataAgt:IsHaveActiveSkillCheats(2) then      --秘籍2：范围延长
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 22 , 20 , 50 * 1.1);
	    LuaDataAgt:MapDrawRectangle(0, 20, 100 * 1.1, 0, 0);
	else
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 22, 20, 50);
	    LuaDataAgt:MapDrawRectangle(0, 20, 100, 0, 0);
	end

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);

	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);

	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList ;
	if LuaDataAgt:IsHaveActiveSkillCheats(2) then      --秘籍2：范围延长
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 22 , 20 , 50 * 1.1);
	else
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 22, 20, 50);
	end

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end


--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	local cnt = targetList.Count;
	for i=0,cnt-1 do
		local target = targetList:get_Item(i);
		target:AddBuff(ThisScriptId, LuaDataAgt, targetList);

		target:AddBuff(544, LuaDataAgt, targetList);

		local TarPos = target:GetCurPos();
		target:PlayFX("Prefabs/BasicEffect/3D_CharEff_XuanYun", target, TarPos, TarPos, 0, true); -- 延迟0播该特效
		target:PlayFX("Prefabs/BasicEffect/3D_CharEff_ZhongDu", target, TarPos, TarPos, 0.8, true); -- 延迟0.8播该特效
	end

	LuaDataAgt:InterruptSiege();
end

GeneralSkill_26 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
}
