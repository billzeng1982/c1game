名称：龙胆虎威
类型：觉醒主动技能
描述：赵云突刺一条直线造成伤害并增加自身暴击并有几率眩晕敌方

local ThisScriptId = 2046;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置技能公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	--LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, -5);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();

	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then
	    local iRange = LuaDataAgt:GetActiveSkillCheatsValue(1)/100 + 1;
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 27.5, 15 , 50);  --- 秘籍1：扩宽范围
	    LuaDataAgt:MapDrawRectangle(0, 15, 125, 0, 0);
	else
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 27.5, 15, 50);  --- 15 = （20 + 部队大小）/ 2
	    LuaDataAgt:MapDrawRectangle(0, 15, 125, 0, 0);
	end
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	--剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	
	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end


	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 27.5, 15, 50);
	
	if (targetList.Count == 0) then
		targetList = LuaDataAgt:FindTarget(TARGET_GROUP.NONE);
	end

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end


--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);

	for i=0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt,VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		
		local targetPos = target:GetCurPos();
		target:PlayFX("Prefabs/SkillEffect/Skill_ZhaoYun_Hit_002", target, targetPos, targetPos, 1, true); -- 延迟0播该特效

		local j = math.random(1,10000);
		if j < 5001 then
		    target:AddBuff(ThisScriptId, LuaDataAgt, targetList);
		end
	end

	-- 添加Buff
	--LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	LuaDataAgt:AddBuff(560, LuaDataAgt, nil);
	local TarPos = LuaDataAgt:GetCurPos();
	LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_BaoJiShangSheng", LuaDataAgt, TarPos, TarPos, 1, true); -- 延迟0播该特效
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillBaseValue(1);
	local MinDmg = Base;
	local Ratio = rSource:GetActiveSkillRateValue(1);
	iRet = Base + (StrAtk - StrDef) * Ratio;
	iRet = math.max(iRet, MinDmg);
	
	--技能吸血60%
	SuckBlood =  iRet * 0.6;
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, SuckBlood,  LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_COMM, 0, 221);
	if not IsOnServer then
		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_HuiFu", LuaDataAgt, TarPos, TarPos, 0.5, true);
		LuaDataAgt:PlayFX("Prefabs/BuffEffect/Buff_JiaXue",  LuaDataAgt, TarPos, TarPos, 0, true);
	end
	local CurHp = rTarget:GetHp(VALUE_TYPE.CURRENT);
	--额外伤害
	local ExtraDamge = (CurHp * 0.2) - StrDef;
	--额外伤害保护
	local ExtraDamge = math.max(ExtraDamge, 0);
	local ExtraDamge = math.min(ExtraDamge,  iRet);
	--之前已做保护
	iRet = iRet + ExtraDamge;
	return -iRet;
end


--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if LuaDataAgt:IsHaveActiveSkillCheats(2) then  --秘籍2： 杀人重置CD
		if iFilterType == FILTER_TYPE.HP then
			local iHp = FilterValue.m_fFoo1;
			if (IsSource and iHp < 0 and
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_GENERALSKILL and FilterValue.m_iValueChgPara == ThisScriptId) then
				-- 本技能打出的伤害
				local target = FilterValue:GetTarget();
				local targetHp = target:GetHp(VALUE_TYPE.CURRENT);
				local Morale = LuaDataAgt:GetActiveSkillBaseValue(2);
				if not IsOnServer and (targetHp + iHp) <= 0 then
					--不在服务器上，敌人被打死，改变我方cd
					LuaDataAgt: SetSkillCD(VALUE_TYPE.CURRENT,0);
				end
			end
		end
	end
	return  FilterValue;
end

GeneralSkill_2046 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
