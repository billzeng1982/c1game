--[[
名称：冲车神行
类型：主动技能
描述：放出一个高速奔驰的冲车，冲向敌军城墙，直接造成伤害
--]]
local ThisScriptId = 10011;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	LuaDataAgt:ClearSkillRange();
	LuaDataAgt:ClearAllSkillTarget();
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	
	-- 再找真正的目标
	LuaDataAgt:FindTargetCity(TARGET_GROUP.ENEMY);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)

	local city = LuaDataAgt:GetTargetCity(TARGET_GROUP.ENEMY)
	city:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		
	
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	local ap = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
	
	if iDamageRef == 1 then
		local iCheatsValue1 = rSource:GetActiveSkillCheatsValue(1);
		
		local atk = rSource:GetActiveSkillAdditionValue(1);
		iRet = atk * iCheatsValue1/100; --加血
	elseif iDamageRef == 2 then
		--秘籍2额外伤害
		local atk = rSource:GetActiveSkillAdditionValue(1);
		local iCheatsValue2 = rSource:GetActiveSkillCheatsValue(2);
		iRet = -(iCheatsValue2/100 + 1) * atk;
	else
		local atk = rSource:GetActiveSkillAdditionValue(1);
		iRet = - atk;
	end
	
	return iRet;
end

GeneralSkill_10011 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
}
