--[[
名称：工事构筑
类型：主动技能
描述：曹仁竖起一道坚不可摧的牢壁，5秒后对其中的敌人造成眩晕和物理伤害。
--]]
local ThisScriptId = 10015;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	LuaDataAgt:MapDrawCircle(0,20);
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 20, 0);
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	if targetList.Count > 1 then
	    return true;
	end
	return false;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 15, 0);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	if iDamageRef == 1 then
		local fCheatsValue = rSource:GetActiveSkillCheatsValue(1);
		local ap = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
		iRet = -ap * fCheatsValue;
	else
		local ad = LuaDataAgt:GetStrAtk(VALUE_TYPE.CURRENT);
		local iTargetStrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
		local stk = LuaDataAgt:GetActiveSkillAdditionValue(1);
		iRet = -stk * (1-iTargetStrDef/(iTargetStrDef+60));
	end
	
	return iRet;
end

GeneralSkill_10015 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
}
