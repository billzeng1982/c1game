--[[
名称：百骑劫营
类型：主动技能
描述：甘宁奇袭敌军，使周围的敌军受到惊吓，大量减速，并且不能使用兵种技。
--]]
local ThisScriptId = 2011;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	LuaDataAgt:ClearTarget();
	--没有技能框 清除其它技能框
	LuaDataAgt:ClearSkillRange();
	--找敌方部队
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 25, 20, 0);
	LuaDataAgt:MapDrawCircle(25,20);
	local targetList2 = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);

	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);

	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end

	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList;
	local targetList2;
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 25, 20, 0);
	targetList2 = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList2);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	local targetList2 = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);

	for i=0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		target:AddBuff(11, LuaDataAgt, nil);
		target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		if not IsOnServer then
			local TarPos = target:GetCurPos();
			target:PlayFX("Prefabs/BasicEffect/3D_CharEff_XuanYun", target, TarPos, TarPos, 0.5, true); -- 延迟0播该特效
			target:PlayFX("Prefabs/SkillEffect/Skill_Xyys_End", target, TarPos, TarPos, 0, true);
		end
	end
	
	for i=0,targetList2.Count-1 do
		local target2 = targetList2:get_Item(i);
		target2:AddBuff(2011, LuaDataAgt, nil);
	end

	LuaDataAgt:InterruptSiege();
end

local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillBaseValue(3);
	local MinDmg = Base;
	local Ratio = rSource:GetActiveSkillRateValue(3);
	iRet = Base + (StrAtk - StrDef) * Ratio;
	iRet = math.max(iRet,MinDmg);

	return -iRet;
end

GeneralSkill_2011 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
}
