--[[
飞骑射日
类型：主动技能
描述：太史慈令全军换上强弓，对周围所有敌军进行射击，持续一段时间。
--]]
local ThisScriptId = 2078;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	--LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_SKILLCHEATS, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt,AIId)
	--没有技能框 清除其它技能框
	LuaDataAgt:ClearSkillRange();
	LuaDataAgt:ClearAllSkillTarget();

	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.ENEMY);

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);

	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);

	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end
	 return true;

end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--设置技能目标为敌方全军
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.ENEMY);

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i = 0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		local sAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.CURRENT);
		local tDef = target:GetStrDef(VALUE_TYPE.CURRENT);
		local k = sAtk / tDef ;
		if k > 1.75 then
		-- 攻是防的1.75倍以上，伤害翻倍
			target:ChgHp(VALUE_TYPE.CURRENT, 1, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		else 
			target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		end
		--概率击晕
		l = target:GetStrAtk(VALUE_TYPE.CURRENT);
		math.randomseed(tostring(os.time() + l):reverse():sub(1, 6)) ;
		local j = math.random(1,100);
		
		local MyRatio = LuaDataAgt:GetActiveSkillBaseValue(2);
		if j < MyRatio then
		   target:AddBuff(569, LuaDataAgt, nil);
		end
	end

	--LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	
	--local TarPos = LuaDataAgt:GetCurPos();
	--LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_BaoJiShangSheng", LuaDataAgt, TarPos, TarPos, 0.5, true); -- 延迟0播该特效
	--[[
	local targetList2 = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	--全队暴击BUFF
	for i=0,targetList2.Count - 1 do
		local target = targetList2:get_Item(i);
		target:AddBuff(ThisScriptId, LuaDataAgt, targetList2);

		local TarPos = target:GetCurPos();
		target:PlayFX("Prefabs/BasicEffect/3D_CharEff_BaoJiShangSheng", target, TarPos, TarPos, 0.5, true); -- 延迟0播该特效
	end
	]]
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = iDamageRef;
	local flag = iDamageRef;
	local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillBaseValue(1);
	local MinDmg = Base;
	local Ratio = rSource:GetActiveSkillRateValue(1);
	
	iRet = Base + (StrAtk - StrDef) * Ratio;
	iRet = math.max(iRet,MinDmg);
	
	if flag == 1 then
	-- 攻是防的两倍以上，伤害翻倍
		iRet = iRet * 2 ;
	end
	
	return -iRet;
end

GeneralSkill_2078 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

	func_doFormula = doFormula;
}
