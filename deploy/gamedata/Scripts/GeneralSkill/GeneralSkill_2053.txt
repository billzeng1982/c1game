--[[
名称：威震逍遥
类型：主动技能
描述：张辽威震逍遥津，大幅提升全队攻击，防御，速度。
--]]
local ThisScriptId = 2053;

--技能诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();

	LuaDataAgt:ClearSkillRange();

	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	local targetList2 = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	LuaDataAgt:MapDrawCircle(0, 30);

	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.SELF);
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	for i = 0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		--加buff加攻加防
		target:AddBuff(53, LuaDataAgt, targetList);

		if not IsOnServer then
			local TarPos =target:GetCurPos();
			target:PlayFX("Prefabs/BasicEffect/3D_CharEff_QuanShuXingSS", target, TarPos, TarPos, 0, true); -- 延迟0播该特效
		end
	end


	LuaDataAgt:ClearTarget();

	local targetList2 = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList2);

	for i = 0,targetList2.Count-1 do
		local target = targetList2:get_Item(i);
		target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		if not IsOnServer then
			local TarPos = target:GetCurPos();
			local strColor = "Blue";
			if not target:IsGroupSelf() then
				strColor = "Red";
			end
			target:PlayFX("Prefabs/SkillEffect/Skill_TY_Hit_"..strColor,  target, TarPos, TarPos, 0, true);
		end
	end
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillBaseValue(3);
	local MinDmg = Base;
	local Ratio = rSource:GetActiveSkillRateValue(3);
	
	iRet = Base + (StrAtk - StrDef) * Ratio;
	iRet = math.max(iRet,MinDmg);

	return -iRet;
end

GeneralSkill_2053 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
}
