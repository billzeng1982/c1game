--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);					--技能的范围
	local targetList_Self = LuaDataAgt:FindTargetEx(TARGET_GROUP.SELF, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);
	
	LuaDataAgt:MapDrawCircle(0,30);

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);

	--没有技能框 清除其它技能框

	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);					--技能的范围
	local targetList_Self = LuaDataAgt:FindTargetEx(TARGET_GROUP.SELF, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)

	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	
	for i = 0,targetList.Count - 1 do
		local target = targetList:get_Item(i);
		--target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		target:AddBuff(ThisScriptId, LuaDataAgt, nil);
		
		local TarPos = target:GetCurPos();
		target:PlayFX("Prefabs/SkillEffect/Skill_Zhtx_Mz_2", target, TarPos, TarPos, 0, true); -- 延迟0播该特效
		target:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongXiaJiang", target, TarPos, TarPos, 0, true);
	end
	
	local targetList_Self = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	for j = 0,targetList_Self.Count - 1 do
		local target = targetList_Self:get_Item(j);
		target:AddBuff(552, LuaDataAgt, nil);
		
		local vPos = target:GetCurPos();
		target:PlayFX("Prefabs/SkillEffect/Skill_Zhtx_Mz_1", target, vPos, vPos, 0, true);
		target:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongShangSheng", target, vPos, vPos, 0, true);
	end;

	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)

	local iRet = 0;

	local WitAtk = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
	local WitDef = rTarget:GetWitDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillBaseValue(3);
	local MinDmg = Base;
	local Ratio = rSource:GetActiveSkillRateValue(3);
	
	iRet = Base + (WitAtk - WitDef) * Ratio;
	iRet = math.max(iRet,MinDmg);
	
	return -iRet;
end

GeneralSkill_81 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

	func_doFormula = doFormula;
}
