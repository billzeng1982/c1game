--[[
名称：八门金锁
类型：主动技能
描述：曹仁竖起一道坚不可摧的牢壁，2.5秒后对其中的敌人造成眩晕和物理伤害。
--]]
local ThisScriptId = 10134;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	
	-- 技能属性（共享值）
	local ap = LuaDataAgt:GetWitAtk(VALUE_TYPE.CURRENT);
	LuaDataAgt:SetShareValue(0, 2.5); -- 八门金锁准备持续时间
	
	LuaDataAgt:SetShareValue(1, 15); -- 八门金锁半径
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	LuaDataAgt:MapDrawCircle(0,20);
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 15, 0);
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	if targetList.Count < 1 then
		return false;
    end
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	if LuaDataAgt:IsHaveActiveSkillCheats(2) or LuaDataAgt:IsHaveActiveSkillCheats(1) then
		local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 15, 0);
	end	
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then
		local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
		for i=0,targetList.Count-1 do
				--立即减血
			   targetList:get_Item(i):ChgHp(VALUE_TYPE.CURRENT, 1, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		end
	end
	if LuaDataAgt:IsHaveActiveSkillCheats(2) then
		local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
		for	i=0,targetList.Count-1 do
			--减速buff
			targetList:get_Item(i):AddBuff(523,LuaDataAgt,nil);
		end
	end
	
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	if iDamageRef == 1 then
		local fCheatsValue = rSource:GetActiveSkillCheatsValue(1);
		local WitAtk = rSource:GetActiveSkillAdditionValue(3);
		iRet = WitAtk * fCheatsValue/100;
		print ("Damege="..iRet);
	else
	    local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	    local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	    local Base = rSource:GetActiveSkillAdditionValue(3);
	    local MinDmg = Base;
	    local Ratio = rSource:GetActiveSkillAdditionValue(4);
	    iRet = Base + (StrAtk - StrDef) * Ratio;
	    iRet = math.max(iRet,MinDmg); 
	end
	
	return -iRet;
end

GeneralSkill_10134 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
}
