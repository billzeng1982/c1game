--[[
名称：抬榇死战
类型：主动技能
描述：扇形AOE，减速，减防
--]]
local ThisScriptId = 49;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--扇形 25是半径 90是直角
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);

	if LuaDataAgt:IsHaveActiveSkillCheats(1) then                   --秘籍1：增大范围
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25 * 1.1, 90);
	    LuaDataAgt:MapDrawFan(25*1.1, -5);
	else
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);
	    LuaDataAgt:MapDrawFan(25, -5);
	end

		--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--扇形 25是半径 90是直角
	local targetList;
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);

	if LuaDataAgt:IsHaveActiveSkillCheats(1) then                   --秘籍1：增大范围
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25 * 1.1, 90);
	else
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);
	end

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i = 0,targetList.Count -1 do
		local target = targetList:get_Item(i);
		-- 伤害
	    target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);

		--减速
		target:AddBuff(ThisScriptId, LuaDataAgt,nil);

		local TarPos = target:GetCurPos();
		target:PlayFX("Prefabs/BasicEffect/3D_CharEff_JianSu", target, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end

	--给自己加防
	--LuaDataAgt:AddBuff(554, LuaDataAgt, nil);
	if not IsOnServer then
		local MyPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuFangShangSheng", LuaDataAgt, MyPos, MyPos, 0, true); -- 延迟0播该特效
	end
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;

	local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	local StrDef1 = rSource:GetStrDef(VALUE_TYPE.CURRENT);
	local StrDef2 = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillBaseValue(1);
	local MinDmg = Base + StrDef1;
	local Ratio = rSource:GetActiveSkillRateValue(1);

	iRet = Base + StrDef1 + (StrAtk - StrDef2) * Ratio;
	iRet = math.max(iRet,MinDmg);

	return -iRet;
end

GeneralSkill_49 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

	func_doFormula = doFormula;
}
