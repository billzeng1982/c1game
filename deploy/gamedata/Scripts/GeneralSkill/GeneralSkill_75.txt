--[[
名称：青龙堰月
类型：主动技能
描述：关羽挥舞偃月刀斩向对手，对前方扇形范围敌军造成伤害，关羽血量越低，伤害越高。
--]]
local ThisScriptId = 75;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end
--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--扇形 25是半径 90是张角
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1)  then 
	    local level = LuaDataAgt:GetActiveSkillCheatsValue(1);
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25 * (1 + level / 100), 90);  --- 秘籍1：增大范围
		targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY,SHAPE_TYPE.CIRCLE, 0, 0, 5 * (1 + level / 100), 0);
	    LuaDataAgt:MapDrawFan(25 * (1 + level / 100), -5 * (1 + level / 100));
	else
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);
	    LuaDataAgt:MapDrawFan(25, -5);
	end
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--扇形 25是半径 90是张角
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1)  then 
	    local level = LuaDataAgt:GetActiveSkillCheatsValue(1);
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25 * (1 + level / 100), 90);  --- 秘籍1：增大范围
		targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5 * (1 + level / 100), 0);
	else
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);
	end
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i = 0,targetList.Count -1 do
		local target = targetList:get_Item(i);
		-- 伤害
	    target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
		
		if LuaDataAgt:IsHaveActiveSkillCheats(2) then  --秘籍2：带易伤
		    target:AddBuff(519,LuaDataAgt,nil);
		end
	end
	
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	local hp = rSource:GetHp(VALUE_TYPE.CURRENT);
	local hpTotal = rSource:GetHp(VALUE_TYPE.RUNTIME_CONF);
	
	local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillAdditionValue(2);
	local MinDmg = Base;
	local Ratio = rSource:GetActiveSkillAdditionValue(3);
	
	iRet = (Base + (StrAtk - StrDef) * Ratio) * (1 + (hpTotal - hp) / hpTotal);
	iRet = math.max(iRet,MinDmg);
	
	return -iRet;
end

GeneralSkill_75 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
}
