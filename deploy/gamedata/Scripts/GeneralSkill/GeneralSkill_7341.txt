名称：天下无双
类型：主动技能

local ThisScriptId = 7341;

--技能诞生处理
local function runBorn(LuaDataAgt)
		-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	--LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_SKILLCHEATS, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 37.5, 15, 75);

    LuaDataAgt:MapDrawCircleRange(0, 30, 0,15,190,0,0);

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);

	--圆形&矩形判断框

	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	--AI操作需要至少命中n人，玩家操作无要求
	--[[
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end
	]]
	
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 30, 0);
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i = 0,targetList.Count-1 do
		local targetHp = targetList:get_Item(i):GetHp(VALUE_TYPE.CURRENT);
		targetList:get_Item(i):ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0); 			--打一次
	end

	LuaDataAgt:ClearTarget();
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 37.5, 15, 75);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	--LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.DISTANCE);
	for i = 0, targetList.Count - 1 do
	    targetList:get_Item(i):ChgHp(VALUE_TYPE.CURRENT, 1, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	end
	
	LuaDataAgt:InterruptSiege();
end


--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = iDamageRef;

	if iFormulaType == VALUE_CHG_TYPE.CHG_HP_SKILLCHEATS then     --秘籍2：吸血，直接返回，防止死循环
	    return iRet;
	end

	local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	local Base0 = rSource:GetActiveSkillBaseValue(1);
	local Base1 = rSource:GetActiveSkillBaseValue(2);
	local MinDmg0 = Base0;
	local MinDmg1 = Base1;
	local Ratio0 = rSource:GetActiveSkillRateValue(1);
	local Ratio1 = rSource:GetActiveSkillRateValue(2);
	if iRet==0 then
		iRet = (Base0 + (StrAtk - StrDef) * Ratio0);
		iRet = math.max(iRet,MinDmg0);
	elseif iRet==1 then
		iRet = (Base1 + (StrAtk - StrDef) * Ratio1);
		iRet = math.max(iRet,MinDmg1);
	end

	if LuaDataAgt:IsHaveActiveSkillCheats(1) then    --秘籍1:20%几率1.5倍伤害
	    math.randomseed(tostring(os.time()+iRet):reverse():sub(1, 6));
	    local i = math.random(1,100);
		if i < 21 then
		    iRet = iRet * 1.5;
		end
	end


	if LuaDataAgt:IsHaveActiveSkillCheats(2) then    --秘籍2：吸血
	    local addBlood = iRet * 0.1;
		print("add .."..addBlood);
		if addBlood > 0 then
		    LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, addBlood, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_SKILLCHEATS, ThisScriptId, 0);
		end
	end

	print ("Damege="..iRet);
	return -iRet;
end


GeneralSkill_7341 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
}
