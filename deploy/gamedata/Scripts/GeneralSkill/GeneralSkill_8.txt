local ThisScriptId = 8;

--技能诞生处理
local function runBorn(LuaDataAgt)

end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--没有技能框 清除其它技能框
	LuaDataAgt:ClearSkillRange();
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);

	--秘籍1：给自己也套盾
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then

	else
		--去掉自己
		LuaDataAgt:RemoveTarget(TARGET_GROUP.SELF, LuaDataAgt)
	end

	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.SELF);
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	--秘籍1：给自己也套盾
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then

	else
		--去掉自己
		LuaDataAgt:RemoveTarget(TARGET_GROUP.SELF, LuaDataAgt)
	end
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);

	--护盾buff
	for i=0,targetList.Count - 1 do
		local target = targetList:get_Item(i);
		target:AddBuff(ThisScriptId, LuaDataAgt, targetList);

		local TarPos = target:GetCurPos();
		target:PlayFX("Prefabs/BasicEffect/3D_CharEff_HuDun", target, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end

	LuaDataAgt:InterruptSiege();
end

GeneralSkill_8 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
}
