描述：庞统用铁锁将周围敌军禁锢起来，使其无法移动，任意一个被锁住的敌军受到伤害时，其余被锁住的敌军都会受到65%的伤害。

local ThisScriptId = 10013;

--技能诞生处理
local function runBorn(LuaDataAgt)
	
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList;
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 35, 0);
    LuaDataAgt:MapDrawCircle(0,35);
	if targetList.Count<2 then
	    return false;
	end
	local count = 0;
	for i=0 , targetList.Count-1 do
		if targetList:get_Item(i):HasBuff(4)  then
			count= count + 1;
		end
	end
	if count > 1 then
		return false;
	end
	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	
	local targetList;
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 35, 0);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	local cnt = targetList.Count;
	for i=0,cnt-1 do
		targetList:get_Item(i):AddBuff(4, LuaDataAgt, targetList);
	end
	
	LuaDataAgt:InterruptSiege();
end

GeneralSkill_10013 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
}
