--[[
名称：吉列先锋
类型：主动技能
描述：自己扣血，接下来的一定时间伤害双倍
--]]
local ThisScriptId = 10025;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList;
	
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.SELF, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);
    LuaDataAgt:MapDrawCircle(0,20);
	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.SELF);

	
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList;
	
	targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.SELF, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local percent = LuaDataAgt:GetActiveSkillAdditionValue(1);
	local hpTotal = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT);
	local iDamage = hpTotal * percent / 100 ;
	
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, -iDamage, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	LuaDataAgt:AddBuff(ThisScriptId,LuaDataAgt,nil);
	
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = iDamageRef;
	
	return iRet;
end

GeneralSkill_10025 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
}
