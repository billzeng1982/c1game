--[[
名称：鬼谋神术
类型：主动技能
描述：郭嘉持续施法
--]]
local ThisScriptId = 48;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置技能公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--没有技能框 清除其它技能框
	LuaDataAgt:ClearSkillRange();
	LuaDataAgt:ClearAllSkillTarget();
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	
	--设置技能目标为自己，表示该技能有目标，才会播放受击特效，生效会延迟
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.NONE);
end


--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	-- 让自己立马停下来
	local currPos = LuaDataAgt:GetCurPos();
	LuaDataAgt:MoveToDest(currPos, true);
	
	-- 添加跳舞Buff
	LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	
	-- 打断攻城
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	local MinDmg = 0;
	local iType = iDamageRef % 10;
	local dancePower = math.floor(iDamageRef / 10);
	
	if iType == 0 then
		local WitAtk = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
		local WitDef = rTarget:GetWitDef(VALUE_TYPE.CURRENT);
		local Base0 = rSource:GetActiveSkillBaseValue(1);
		local Ratio0 = rSource:GetActiveSkillRateValue(1);
		MinDmg = Base0;
		
		iRet = (Base0 + (WitAtk - WitDef) * Ratio0);
		
	elseif iType == 1 then
		local Base1 = rSource:GetActiveSkillBaseValue(2);
		MinDmg = Base1;
		
		iRet = Base1;
	end
	
	--跳舞层数成100%
	if dancePower >= 4 then
		local DanceAddRatio = rSource:GetArmyBaseValue();
		iRet = iRet * DanceAddRatio;	
	end
	
	if rSource:IsHaveActiveSkillCheats(2) then    --秘籍2:10%几率1.5倍伤害
	    math.randomseed(tostring(os.time()+iRet):reverse():sub(1, 6));
	    local i = math.random(1,100);
		if i < 11 then
		    iRet = iRet * 1.5;
		end
	end
	
	iRet = math.max(iRet, MinDmg);
	return -iRet;
end

GeneralSkill_48 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
}
