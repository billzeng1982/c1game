--[[
名称：狱火焚城
类型：主动技能
描述：扇形法术伤害技，可以打城墙
--]]
local ThisScriptId = 10107;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--扇形 25是半径 90是张角
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1)  then 
	    local level = LuaDataAgt:GetActiveSkillCheatsValue(1);
		LuaDataAgt:FindTargetCityEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25*(1+level/100), 90);  -- 找城墙
		
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25*(1+level/100), 90);  --- 秘籍1：增大范围
		targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY,SHAPE_TYPE.CIRCLE, 0, 0, 5*(1+level/100), 0);
	    LuaDataAgt:MapDrawFan(25*(1+level/100), -5*(1+level/100));
	else
		LuaDataAgt:FindTargetCityEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);  -- 找城墙
	
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);
	    LuaDataAgt:MapDrawFan(25, -5);
	end
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	
	if targetList.Count < 1 then
	    return false;
	end
	
	--出城时间判定，出城后5秒内不允许放技能
	local passedTime = LuaDataAgt:GetOutCitySuccessTimePassed();
	if passedTime < 5 then
	    return false;
	end
	 
	
	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--扇形 25是半径 90是张角
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1)  then 
	    local level = LuaDataAgt:GetActiveSkillCheatsValue(1);
		LuaDataAgt:FindTargetCityEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25*(1+level/100), 90);  -- 找城墙
		
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25*(1+level/100), 90);  --- 秘籍1：增大范围
		targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5*(1+level/100), 0);
	else
		LuaDataAgt:FindTargetCityEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);  -- 找城墙
	    LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.FAN, 0, 0, 25, 90);
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 5, 0);
	end
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i = 0,targetList.Count -1 do
		-- 伤害
	    targetList:get_Item(i):ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	end
	
	local city = LuaDataAgt:GetTargetCity(TARGET_GROUP.ENEMY)
	if city ~= nil then
		city:ChgHp(VALUE_TYPE.CURRENT, 1, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	end
	
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	if (iDamageRef == 0) then
		-- 打人伤害
		local WitAtk = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
		local WitDef = rTarget:GetWitDef(VALUE_TYPE.CURRENT);
		local Base = rSource:GetActiveSkillBaseValue(1);
		local MinDmg = Base;
						print ("Base="..Base);
		local Ratio = rSource:GetActiveSkillRateValue(1);
				print ("Ratio="..Ratio);
		iRet = (Base + (WitAtk - WitDef) * Ratio);
		iRet = math.max(iRet,MinDmg);
		print ("Damege="..iRet);
	else
		-- 打城墙伤害
		local WallBase = rSource:GetActiveSkillBaseValue(2);
		iRet = WallBase;
		iRet = math.max(iRet, 50);
	end
	
	return -iRet;
end

GeneralSkill_10107 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
}
