类型：主动技能
描述：张飞爆发出一声怒喝，对周围敌军造成少量伤害，附加减速和易伤。

local ThisScriptId = 10111;

--技能诞生处理
local function runBorn(LuaDataAgt)
	
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1)  then 
	    local level = LuaDataAgt:GetActiveSkillCheatsValue(1);
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 22*(1+level/100), 25*(1+level/100), 0);  --- 秘籍1：增大范围
	    LuaDataAgt:MapDrawCircle(22*(1+level/100),25*(1+level/100));
	else
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 22, 25, 0);
		LuaDataAgt:MapDrawCircle(22,25);
	end
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);		 
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	
	--圆形判断框
	
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	
	if targetList.Count < 1 then
	    return false;
	end
	
	--出城时间判定，出城后5秒内不允许放技能
	local passedTime = LuaDataAgt:GetOutCitySuccessTimePassed();
	if passedTime < 5 then
	    return false;
	end
	
	--当前位置判定，在城墙下方的一段区域内不允许放技能
	local pos = LuaDataAgt:GetCurPos();
	if pos.z < 8  or  pos.z > 72 then 
	    return false;
	end 
	
	
	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1)  then 
	    local level = LuaDataAgt:GetActiveSkillCheatsValue(1);
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 22*(1+level/100), 25*(1+level/100), 0);  --- 秘籍1：增大范围
	else
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 22, 25, 0);
	end
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i = 0,targetList.Count-1 do
		--加buff
		targetList:get_Item(i):AddBuff(14, LuaDataAgt, targetList);
		targetList:get_Item(i):AddBuff(537, LuaDataAgt, targetList);
	end
		-- 添加自己加速
	LuaDataAgt:AddBuff(538, LuaDataAgt, nil);
	LuaDataAgt:InterruptSiege();
end


GeneralSkill_10111 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

}
