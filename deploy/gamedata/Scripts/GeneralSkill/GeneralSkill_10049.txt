--[[
名称：悬壶济世 
类型：主动技能
描述：华佗以仁义之心救济世人、妙手回春，使我方血量最低的部队回复大量生命并且增加物攻物防。
--]]
local ThisScriptId = 10049;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--没有技能框 清除其它技能框
	LuaDataAgt:ClearSkillRange();
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	
	if targetList.Count > 0 then
	    --目标是己方血最少的
		LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.HP);
		local percent = targetList:get_Item(0):GetHp(VALUE_TYPE.CURRENT) / targetList:get_Item(0):GetHp(VALUE_TYPE.ORIGIN_CONF);
		if  percent < 0.6 then
		    return true;
		end
	end
	
	return false;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	
	--目标是己方血最少的
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.HP);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)

	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	
	if targetList.Count > 0 then
		--只加血
		targetList:get_Item(0):ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
	end
	
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	if iDamageRef == 1 then
		local value = rSource:GetActiveSkillCheatsValue(2);
		local targetHp = rTarget:GetHp(VALUE_TYPE.ORIGIN_CONF);
		local stk = rSource:GetActiveSkillAdditionValue(1);
		iRet = targetHp * value / 100;
	else
		local ap = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
		local stk = rSource:GetActiveSkillAdditionValue(1);
		iRet = rTarget:GetHp(VALUE_TYPE.ORIGIN_CONF) * stk / 100;
	end
	
	return iRet;
end

GeneralSkill_10049 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
}
