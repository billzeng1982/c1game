--技能条件判断
local function checkCond(LuaDataAgt,AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 21, 12, 32);

	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);

	--目标是对面距离最近的
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.DISTANCE);

	--没有技能框 清除其它技能框
	LuaDataAgt:MapDrawRectangle(0,12,64,0,0);
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);

	--AI操作需要至少命中n人，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			return true;
		else
			return false;
		end
	end
	return true;

end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 21, 12, 32);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	--目标是对面距离最近的
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.DISTANCE);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i=0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		--嘲讽buff
		target:AddBuff(ThisScriptId, LuaDataAgt, targetList);

		--击杀回血buff
		target:AddBuff(558, LuaDataAgt, nil);
		if not IsOnServer then
			target:PlayFX("Prefabs/BasicEffect/3D_CharEff_ChaoFeng", target, Vector3.zero, Vector3.zero, 0, true); -- 延迟0播该特效
			target:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuFangXiaJiang", target, Vector3.zero, Vector3.zero, 0.5, true); -- 延迟0播该特效
			target:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangXiaJiang", target, Vector3.zero, Vector3.zero, 1, true); -- 延迟0播该特效
		end
	end

	--许褚自加buff
	LuaDataAgt:AddBuff(539, LuaDataAgt, nil);
	if not IsOnServer then
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongShangSheng", LuaDataAgt, Vector3.zero, Vector3.zero, 0, true); -- 延迟0播该特效
	end
	LuaDataAgt:InterruptSiege();
end

GeneralSkill_52 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
}
