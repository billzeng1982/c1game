--[[
名称：悬壶济世
类型：主动技能
描述：华佗以仁义之心救济世人、妙手回春，使我方血量最低的部队回复大量生命并且增加物攻物防。
--]]
local ThisScriptId = 2007;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--没有技能框 清除其它技能框
	LuaDataAgt:ClearSkillRange();
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	local targetList2 = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	LuaDataAgt:RemoveTroopInVisible(targetList2);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	--目标是我方血最少的
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.HP);

	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.SELF);

	--AI操作需要至少有一个血量低于50%的目标，玩家操作无要求
	if (AIId == 0) then

	else
		if targetList.Count > 0 then
			local target = targetList:get_Item(0);
			local maxHp = target:GetHp(VALUE_TYPE.RUNTIME_CONF);
			local curHp = target:GetHp(VALUE_TYPE.CURRENT);
			if curHp / maxHp < 0.5 then
				return true;
			else
				return false;
			end
		else
			return false;
		end
	end

	if targetList.Count > 0 then
		return true;
	end

	LuaDataAgt:SetSkillPrompt(5);
	return false;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	--找我方部队
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	--目标是己方血最少的
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.HP);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	
	--给血最低的加血
	if targetList.Count > 0 then
		--给血最低的加血
		targetList:get_Item(0):ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 221);
	end
	
	
	--给全体加血
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList2 = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList2);

	for i=0,targetList2.Count-1 do
		local target = targetList2:get_Item(i);
		target:ChgHp(VALUE_TYPE.CURRENT, 1, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 221);
		
		if not IsOnServer then
			target:PlayFX("Prefabs/BuffEffect/Buff_JiaXue", target, Vector3.zero, Vector3.zero, 1, true);
			--target:PlayFX("Prefabs/BasicEffect/3D_CharEff_HuiFu", target, Vector3.zero,  Vector3.zero, 1.5, true);
		end
	end

	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = iDamageRef;

	if iDamageRef == 0 then
		local WitAtk = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
		local Base = rSource:GetActiveSkillBaseValue(1);
		local Ratio = rSource:GetActiveSkillRateValue(1);
		iRet = Base + (WitAtk) * Ratio;

		return iRet;
	end

	if iDamageRef == 1 then
		local WitAtk = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
		local Base = rSource:GetActiveSkillBaseValue(1);
		local Ratio = rSource:GetActiveSkillRateValue(1);
		iRet = Base + (WitAtk) * Ratio;
		iRet = iRet/2;

		return iRet;
	end
end

GeneralSkill_2007 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

	func_doFormula = doFormula;
}
