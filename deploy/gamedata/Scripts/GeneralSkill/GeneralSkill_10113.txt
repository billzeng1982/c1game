--[[
名称：炽炎火舟
类型：主动技能
描述：扣自己血，打击并击晕敌人
--]]
local ThisScriptId = 10113;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 62, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt, AIId)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then                   --秘籍1：范围增长
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 17.5, 15, 35*1.1);  --- 15 = （20 + 部队大小）/ 2
	    LuaDataAgt:MapDrawRectangle(0,15,80*1.1,0,0);
	else
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 17.5, 15, 35);  --- 15 = （20 + 部队大小）/ 2
	    LuaDataAgt:MapDrawRectangle(0,15,80,0,0);
	end
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	 
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	
	--目标是对面距离最近的
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.DISTANCE);
	
	--小地图技能目标提示标志
	LuaDataAgt:SetMapSkillTarget(TARGET_GROUP.ENEMY);
	
	if targetList.Count > 0 then
		return true;
	end
	
	LuaDataAgt:SetSkillPrompt(5);
	return false;

end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
	local targetList;
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then                   --秘籍1：范围增长
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 17.5, 15, 35*1.1);  --- 15 = （20 + 部队大小）/ 2
	else
	    targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.RECT, 0, 17.5, 15, 35);  --- 15 = （20 + 部队大小）/ 2
	end
	
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	-- 剔除不可见部队
	LuaDataAgt:RemoveTroopInVisible(targetList);
	--目标是对面距离最近的
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.DISTANCE);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, 2, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 62, 0);
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.ENEMY);
	for i = 0 , targetList.Count - 1 do
	    targetList:get_Item(i):ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 62, 0);	
		--给这个人加个BUFF，BUFF效果是让自己周围友军掉血
	    targetList:get_Item(i):AddBuff(62, LuaDataAgt, nil);	
	end
	LuaDataAgt:InterruptSiege();
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	if iDamageRef == 0 then
			local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	        local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	        local Base = rSource:GetActiveSkillBaseValue(2);
	        local MinDmg = Base;
	        local Ratio = rSource:GetActiveSkillRateValue(2);
	        iRet = Base + (StrAtk - StrDef) * Ratio;
	        iRet = math.max(iRet,MinDmg);
	        print ("Damege 1="..iRet);

	elseif iDamageRef == 1 then
	        local StrAtk = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
	        local StrDef = rTarget:GetStrDef(VALUE_TYPE.CURRENT);
	        local Base = rSource:GetActiveSkillBaseValue(3);
	        local MinDmg = Base;
	        local Ratio = rSource:GetActiveSkillRateValue(3);
	        iRet = (Base + (StrAtk - StrDef) * Ratio );
	        iRet = math.max(iRet,MinDmg);
			if LuaDataAgt:IsHaveActiveSkillCheats(2) then           --秘籍2：溅射伤害提高
			    iRet = iRet * 1.3;
			end
	        print ("Damege 2="..iRet);
	elseif iDamageRef == 2 then
			local iRet = rSource:GetActiveSkillBaseValue(1);
			print ("Damege self="..iRet);
	end

	return -iRet;
end

GeneralSkill_10113 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFilter = doFilter;
	func_doFormula = doFormula;
	
}
