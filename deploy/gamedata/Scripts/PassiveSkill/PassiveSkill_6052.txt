--技能生效，数值处理
local function takeEffect(LuaDataAgt)
if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	local ac = 0;
	local shield = 0;
	local spear = 0;
	local cavalry = 0;
	local wiser = 0;
	local taoist= 0;
	--计算兵种个数
	for i = 0,targetList.Count-1 do
		local armytypecheck=targetList:get_Item(i):GetArmyType();
		if (armytypecheck== ARMY_TYPE.ARMY_ARCHER) then
			ac = 1;
		elseif (armytypecheck== ARMY_TYPE.ARMY_SPEARMAN) then
			spear = 1;
		elseif (armytypecheck== ARMY_TYPE.ARMY_CAVALRY) then
			cavalry = 1;
		elseif (armytypecheck== ARMY_TYPE.ARMY_WISER) then
			wiser = 1;
		elseif (armytypecheck== ARMY_TYPE.ARMY_SHIELDMAN) then
			shield = 1;
		elseif (armytypecheck== ARMY_TYPE.ARMY_TAOIST) then
			taoist = 1;
		end
	end
	
	--给各兵种加加强版buff
	for i = 0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		local armytypecheck = target:GetArmyType();
		if (armytypecheck== ARMY_TYPE.ARMY_ARCHER or 
			armytypecheck== ARMY_TYPE.ARMY_SPEARMAN or 
			armytypecheck== ARMY_TYPE.ARMY_CAVALRY or
			armytypecheck== ARMY_TYPE.ARMY_TAOIST) then
			if ac + spear+ cavalry+wiser+shield+taoist > 3 then
				target:AddBuff(13013, LuaDataAgt, targetList);
			else
				target:AddBuff(ThisScriptId, LuaDataAgt, targetList);
			end
		end
	end
	
	LuaDataAgt:SetSkillEnable(false);
end
