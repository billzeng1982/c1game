--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
		
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	LuaDataAgt:RemoveNotMiniTarget(targetList, CURRENT_VALUE_TYPE.WITATK);
	
	local chgValue = LuaDataAgt: GetPassiveSkillBaseValue(ThisScriptId,1);
	if targetList.Count > 0 then
		targetList:get_Item(0):ChgWitDef(VALUE_TYPE.CURRENT,chgValue);
	end
	
	LuaDataAgt:SetSkillEnable(false);
end
