local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	LuaDataAgt:SetSkillEnable(false);
end

local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType ~= VALUE_CHG_TYPE.CHG_HP_GENERALSKILL then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and not IsSource and not IsOnServer then
			local Timer = LuaDataAgt:GetShareTimer(0);
			if Timer < 0 then
				LuaDataAgt:SetShareTimer(0, 1);
				LuaDataAgt:AddBuff( 6492, LuaDataAgt, nil);
				local iTime = LuaDataAgt:GetShareTimer(1);
				if iTime < 0 then
					local TarPos = LuaDataAgt:GetCurPos();
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuFangShangSheng", LuaDataAgt, TarPos, TarPos, 0, true);
					LuaDataAgt:SetShareTimer(1,8);
				end
			end
		end
	end

	return FilterValue;
end
