local function takeEffect(LuaDataAgt)

--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.TRIG then
		if (FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_TRIG_TROOP_OUTCITY_DONE or 
			FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_TRIG_TROOP_DEAD_DONE or 
			FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_TRIG_TROOP_RETREAT_DONE) then
			if LuaDataAgt:IsOnFieldOnlyMe() then
				LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
			else
				LuaDataAgt:DelBuff(ThisScriptId, BUFF_TYPE.BUFF_TYPE_NONE, LuaDataAgt);
			end
		end
	end
	
	return  FilterValue;
end
