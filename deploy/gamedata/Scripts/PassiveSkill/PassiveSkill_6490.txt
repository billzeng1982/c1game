--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 被动技能一场比赛只生效一次
	LuaDataAgt:SetSkillEnable(false);
end

end

--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_NORMAL then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource and not IsOnServer then
			local iValueStune = FilterValue:GetSource():GetPassiveSkillBaseValue(ThisScriptId,1)*100;
			print("iValueStune=="..iValueStune);
			local j = math.random(1,10000);

			if j < iValueStune then
				local target = FilterValue:GetTarget();
				target:AddBuff(ThisScriptId, LuaDataAgt, nil);
			end
		end
	end
end

--这个ID要和脚本ID保持一致

	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
	func_doFilter = doFilter;
