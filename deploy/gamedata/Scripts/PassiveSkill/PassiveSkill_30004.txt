--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	local maxHp = LuaDataAgt:GetHp(VALUE_TYPE.RUNTIME_CONF);
	local curHp = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT);
	if curHp <= maxHp * 0.2 then
	   LuaDataAgt:DelBuff(0, BUFF_TYPE.BUFF_TYPE_ATTR_LOSS, LuaDataAgt);
	   LuaDataAgt:DelBuff(0, BUFF_TYPE.BUFF_TYPE_SPEED_LOSS, LuaDataAgt);
	   LuaDataAgt:DelBuff(0, BUFF_TYPE.BUFF_TYPE_FAINT, LuaDataAgt);
	   LuaDataAgt:DelBuff(0, BUFF_TYPE.BUFF_TYPE_RESTRAIN, LuaDataAgt);
	   LuaDataAgt:DelBuff(0, BUFF_TYPE.BUFF_TYPE_SCOFF, LuaDataAgt);
	end
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 一场战斗中只生效一次
	LuaDataAgt:SetSkillEnable(false);
end

--Skill过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if not IsSource and iHp < 0 then
			if not IsOnServer then
				local maxHp = LuaDataAgt:GetHp(VALUE_TYPE.RUNTIME_CONF);
				local curHp = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT);
				if curHp <= maxHp * 0.8 then
					LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
				end
			end
		end	
	end
	
	return FilterValue;
end
