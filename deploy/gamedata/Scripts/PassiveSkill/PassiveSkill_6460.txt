--[[
名称:赵云被动1
类型:被动技能
描述:每击退一个部队，回复一定血量
--]]
local ThisScriptId = 6460; 

--技能诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, -5);
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_PASSIVESKILL, ThisScriptId, 0);

	print("RunBorn PassiveSkill Id = "..ThisScriptId);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象,上面条件满足后执行
local function findTarget(LuaDataAgt)
	
end

--技能生效,数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 被动技能一场比赛只生效一次
	LuaDataAgt:SetSkillEnable(false);
end

--Skill过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if (IsSource and iHp < 0 ) then
			-- 我打出的伤害
			local target = FilterValue:GetTarget();
			local targetHp = target:GetHp(VALUE_TYPE.CURRENT);
			
			if (targetHp + iHp) <= 0 then
				-- 不在服务器上，敌人被打死，回血
				LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_PASSIVESKILL, ThisScriptId, 221);
				if not IsOnServer then 
					local MyPos = LuaDataAgt:GetCurPos();
					--LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_HuiFu", LuaDataAgt, MyPos, MyPos, 0, true); -- 延迟0播该特效	
					LuaDataAgt:PlayFX("Prefabs/BuffEffect/Buff_JiaXue",  LuaDataAgt, MyPos, MyPos, 0, true);
				end
			end
		end
	end
	
	return  FilterValue;
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local Ratio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local MaxHp = LuaDataAgt:GetHp(VALUE_TYPE.RUNTIME_CONF);
	
	local iRet = MaxHp * Ratio;
	return iRet;
end

--这个ID要和脚本ID保持一致
PassiveSkill_6460 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
