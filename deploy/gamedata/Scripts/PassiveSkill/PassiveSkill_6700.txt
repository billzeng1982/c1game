--技能生效,数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	
	-- 被动技能一场比赛只生效一次
	LuaDataAgt:SetSkillEnable(false);
end

--Skill过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	local Source = FilterValue:GetSource();

	if iFilterType == FILTER_TYPE.TRIG and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_TRIG_BUFF_ADD_END then
		
		if not IsSource then
			if LuaDataAgt:HasBuff(9998) then
				--如果有冲击波Buff，无视控制
				local iBuffType = BUFF_TYPE.IntToEnum(FilterValue.m_fFoo2);
				if (	iBuffType == BUFF_TYPE.BUFF_TYPE_SPEED_LOSS or
					iBuffType == BUFF_TYPE.BUFF_TYPE_FAINT or
					iBuffType == BUFF_TYPE.BUFF_TYPE_RESTRAIN or
					iBuffType == BUFF_TYPE.BUFF_TYPE_SCOFF) then
					FilterValue.m_fFoo1 = 0;
				end
			end
		end
	end
	
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_RUSH then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource and not IsOnServer then
			LuaDataAgt:AddBuff(13022, LuaDataAgt, nil);
		end	
	end
	
	return FilterValue;
end
	
--这个ID要和脚本ID保持一致

	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
