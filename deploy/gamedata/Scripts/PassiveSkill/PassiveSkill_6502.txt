local function takeEffect(LuaDataAgt)
--技能生效，数值处理
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	--计算魏国武将个数
	local iCnt = 0;
	for i = 0,targetList.Count-1 do
		local checkcountryid = targetList:get_Item(i):GetCountryID();
		if (checkcountryid == 1) then
			iCnt = iCnt + 1;
		end
	end
	
	for i = 0,targetList.Count-1 do
		local checkcountryid = targetList:get_Item(i):GetCountryID();
		if (checkcountryid == 1) then
			if iCnt > 3 then
				targetList:get_Item(i):AddBuff(13011, LuaDataAgt, targetList);
			else
				targetList:get_Item(i):AddBuff(ThisScriptId, LuaDataAgt, targetList);
			end
		end
	end
	
	LuaDataAgt:SetSkillEnable(false);
end
