local ThisScriptId = 6720;

--技能诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_PASSIVESKILL, ThisScriptId, 0);

	print("RunBorn PassiveSkill Id = "..ThisScriptId);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象,上面条件满足后执行
local function findTarget(LuaDataAgt)

end

--技能生效,数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end

	LuaDataAgt:SetSkillEnable(false);
end


local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SPEAR then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource then
			LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, -iHp,  LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_PASSIVESKILL, ThisScriptId, 221);
			
			if not IsOnServer then
				local TarPos = LuaDataAgt:GetCurPos();
				--LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_HuiFu", LuaDataAgt, TarPos, TarPos, 1.5, true);
				LuaDataAgt:PlayFX("Prefabs/BuffEffect/Buff_JiaXue",  LuaDataAgt, TarPos, TarPos, 1, true);
			end
		end
	end
	
	return FilterValue;
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local SuckBloodRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId,1) / 100;
	iRet = iDamageRef * SuckBloodRatio;
	
	return iRet;
end

--这个ID要和脚本ID保持一致
PassiveSkill_6720 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
