	-- 添加血量过滤处理
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
	
	print("RunBorn PassiveSkill Id = "..ThisScriptId);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	
end

--技能生效，数值处理（update调用 takeEffect）
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end

	--找高顺和董卓
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	local gs = nil;
	local dz = nil;
	for i = 0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		local check = target:GetGeneralId();
		if check == 72 then
			dz = target;
		elseif check ==83 then
			gs = target;
		end
	end

	if (gs ~= nil and dz ~= nil) then
		gs:AddBuff(ThisScriptId, LuaDataAgt, nil);
		dz:AddBuff(ThisScriptId, LuaDataAgt, nil);
		LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	end

	LuaDataAgt:SetSkillEnable(false);
end

--Skill过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SPEAR then
		local iHp = FilterValue.m_fFoo1;
		if IsSource and iHp < 0 then
			local OutSpear = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100; --肉搏加成
			FilterValue.m_fFoo1 = iHp * (1 + OutSpear);
		end
	end
	
	return FilterValue;
end

PassiveSkill_6682 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

	func_doFilter = doFilter;
}
