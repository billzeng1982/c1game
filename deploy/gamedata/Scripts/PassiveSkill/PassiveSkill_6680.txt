--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 加减属性
	-- 添加血量过滤处理
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
	
	LuaDataAgt: SetShareTimer(1,0);
	
	print("RunBorn PassiveSkill Id = "..ThisScriptId);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 被动技能一场比赛只生效一次
	LuaDataAgt:SetSkillEnable(false);
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	
end

--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SPEAR then
	    local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource and not IsOnServer then
			local Chance = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
			
			math.randomseed(os.time());
			local randomValue = math.random();
			
			if randomValue <= Chance then
				local Timer = LuaDataAgt: GetShareTimer(1);
				if Timer < 0 then
					local target = FilterValue:GetTarget();
					
					target:ChgMorale(-1);
					--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_ShiQiXiaJiang", 0.5, target:GetCurPos());
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_ShiQiXiaJiang", LuaDataAgt, target:GetCurPos(), target:GetCurPos(), 0.5, true);
					--20秒内只能触发一次
					LuaDataAgt: SetShareTimer(1,20);
				end
			end
		end
	end
end

--这个ID要和脚本ID保持一致
PassiveSkill_6680 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
