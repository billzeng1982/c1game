	return true;
end

local function findTarget(LuaDataAgt)

end

--技能生效,数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	LuaDataAgt:SetSkillEnable(false);
end

-- 过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_NORMAL then
		local iHp = FilterValue.m_fFoo1;
		if 	iHp < 0 and IsSource then
			
			local source = FilterValue:GetSource();
			local target = FilterValue:GetTarget();
			local sourcePos = source:GetCurPos();
			local targetPos = target:GetCurPos();

			local Chance = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) * 100;
			local j = math.random(1,10000);

			if j < Chance  then
				FilterValue.m_fFoo1 = 2 * iHp;

				-- 表现
				if not IsOnServer then
					local strColor = "Blue";
					if not target:IsGroupSelf() then
						strColor = "Red";
					end

					LuaDataAgt:PlayFX("Prefabs/PassiveSkillEffect/LiuBei/Skill_LiuBei_002", LuaDataAgt, sourcePos, targetPos, 0, true);

					if target:GetArmyType() == ARMY_TYPE.ARMY_CRASHCAR then
						--粮车、冲车、难民
						LuaDataAgt:PlayFX("Prefabs/BasicEffect/NormalAtkFx_Che", target, targetPos, targetPos, 0, true);
					else
						LuaDataAgt:PlayFX("Prefabs/BasicEffect/FX_Hit_More_" .. strColor, target, targetPos, targetPos, 0.15, true);
						LuaDataAgt:PlayFX("Prefabs/BasicEffect/FX_Hit_More_" .. strColor, target, targetPos, targetPos, 0.35, true);
					end
				end

			end
		end
	end

	return FilterValue;
end


--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	iRet = iDamageRef;
	return -iRet;
end

PassiveSkill_6150 = {

	func_findTarget = findTarget;
	func_takeEffect = takeEffect;

	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
