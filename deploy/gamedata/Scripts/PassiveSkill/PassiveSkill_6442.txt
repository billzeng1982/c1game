名称:三国无双
类型:被动技能
描述:增加自身伤害，减少自身受到的伤害，如果友方武将势力包含4个势力则效果翻倍
--]]
local ThisScriptId = 6442; 


--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 设置公式
	--LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_PASSIVESKILL, ThisScriptId, 0);
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, -1);
	--计算势力个数
	local totalCount;
	local checkcountryid;
	local checknum1 = 0;
	local checknum2 = 0;
	local checknum3 = 0;
	
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	--自己不算
	for i = 0,targetList.Count-1 do
		checkcountryid = targetList:get_Item(i):GetCountryID();
		if (checkcountryid == 1) then
			checknum1 = 1;
		elseif (checkcountryid == 2) then
			checknum2 = 1;
		elseif (checkcountryid == 3) then
			checknum3 = 1;
		end
	end
	totalCount = checknum1 + checknum2 + checknum3 ; 
	LuaDataAgt:SetShareValue(0,totalCount);
	
	print("RunBorn PassiveSkill Id = "..ThisScriptId);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	local totalCount = LuaDataAgt:GetShareValue(0);
	
	if (totalCount == 3) then
		LuaDataAgt:AddBuff(13009, LuaDataAgt, nil);
	else
		LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	end
	
	LuaDataAgt:SetSkillEnable(false);
end


--Skill过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	local OutAll = LuaDataAgt:GetPassiveSkillBaseValue(6442,1)/100;

	local totalCount = LuaDataAgt:GetShareValue(0);
	if iFilterType == FILTER_TYPE.HP then
		--输出伤害处理
		if IsSource and FilterValue.m_fFoo1 < 0 then
			--所有，不包含栅栏	
			if not (FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_TOWER or
					FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_BARRIER) then
				local iHp = FilterValue.m_fFoo1;
				if (totalCount == 3) then
					local OutAll2 = OutAll * 2
					FilterValue.m_fFoo1 = iHp * (1 + OutAll2);
				else
					FilterValue.m_fFoo1 = iHp * (1 + OutAll);
				end 
			end			
		end
	end
	return FilterValue;
end
--这个ID要和脚本ID保持一致
PassiveSkill_6442 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
