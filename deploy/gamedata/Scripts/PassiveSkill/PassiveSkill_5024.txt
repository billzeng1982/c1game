--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 改变万箭齐发所耗士气值
	LuaDataAgt:ChgActiveSkillMorale(VALUE_TYPE.CURRENT, -1);
	
	-- 设置共享值（万箭齐发半径）
	local curRadius = LuaDataAgt:GetActiveSkillShareValue(0);
	LuaDataAgt:SetActiveSkillShareValue(0, curRadius * 1.15);
	
	-- 设置共享值（万箭齐发持续时间）
	local curDuration = LuaDataAgt:GetActiveSkillShareValue(1);
	LuaDataAgt:SetActiveSkillShareValue(1, curDuration + 2);
	
	-- 一场战斗中只生效一次
	LuaDataAgt:SetSkillEnable(false);
end
