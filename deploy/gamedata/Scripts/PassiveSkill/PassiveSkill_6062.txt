--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	local checkcountryid;
	local checknum1 = 0;
	local checknum2 = 0;
	local checknum3 = 0;
	
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	--计算势力个数(自己不算了)
	for i = 0,targetList.Count-1 do
		checkcountryid = targetList:get_Item(i):GetCountryID();
		if (checkcountryid == 1) then
			checknum1 = 1;
		elseif (checkcountryid == 2) then
			checknum2 = 1;
		elseif (checkcountryid == 3) then
			checknum3 = 1;
		end
	end
	
	if (checknum1+checknum2+checknum3 == 3) then
		LuaDataAgt:AddBuff(13007, LuaDataAgt, nil);
	else
		LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	end
	
	LuaDataAgt:SetSkillEnable(false);
end
