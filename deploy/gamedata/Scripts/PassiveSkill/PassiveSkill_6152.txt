--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	--计算蜀国武将个数
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	local iCnt = 0;
	for i = 0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		local check = target:GetCountryID();
		if check == 2 then
			iCnt = iCnt + 1;
		end
	end

	for i = 0,targetList.Count-1 do
		local target = targetList:get_Item(i);
		local check = target:GetCountryID();
		if check == 2 then
			if iCnt >= 4 then
				target:AddBuff(13008, LuaDataAgt, nil);
			else
				target:AddBuff(ThisScriptId, LuaDataAgt, nil);
			end
		end
	end
	
	LuaDataAgt:SetSkillEnable(false);
end

end
