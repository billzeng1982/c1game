local ThisScriptId = 6790; 


--技能诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);	
	print("RunBorn PassiveSkill Id = "..ThisScriptId);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象,上面条件满足后执行
local function findTarget(LuaDataAgt)
	
end

--技能生效,数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(true);
		return;
	end
end

local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_RUSH then
	    local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource then
			local target = FilterValue:GetTarget();
			local maxHp = target:GetHp(VALUE_TYPE.RUNTIME_CONF);
			local curHp = target:GetHp(VALUE_TYPE.CURRENT);
			if curHp < 0.45 * maxHp then
				local Ratio = LuaDataAgt: GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
				iHp = iHp * (1 + Ratio);
				FilterValue.m_fFoo1 = iHp;
				
				if not IsOnServer then
					--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_EWaiShangHai", 1 , LuaDataAgt:GetCurPos());
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_EWaiShangHai", LuaDataAgt, LuaDataAgt:GetCurPos(), LuaDataAgt:GetCurPos(), 1, true);
					if curHp + iHp <= 0 then
						-- 不在服务器上，敌人被打死，改变我方士气
						LuaDataAgt:ChgMorale(1);
					end
				end
			end
		end
	end
	
	return FilterValue;
end

--这个ID要和脚本ID保持一致
PassiveSkill_6790= {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_doFilter = doFilter;
	
}
