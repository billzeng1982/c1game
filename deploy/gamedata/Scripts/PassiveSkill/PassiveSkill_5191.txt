--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	-- 添加血量过滤处理（团队过滤）
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.ENEMY, 0);
	
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 一场战斗中只生效一次
	LuaDataAgt:SetSkillEnable(false);
end

--Skill过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP then
		if  FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_CITYBACK then
			local iHp = FilterValue.m_fFoo1;
			print("iHp="..iHp);
			FilterValue.m_fFoo1 = iHp * 1.15;
		end
	end
	
	return FilterValue;
end
