--技能生效，数值处理
local function takeEffect(LuaDataAgt)
if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	
	local ac = 0;
	local shield = 0;
	local spear = 0;
	local cavalry = 0;
	local wiser = 0;
	--计算兵种个数
	for i = 0,targetList.Count-1 do
		local armytypecheck=targetList:get_Item(i):GetArmyType();
		if (armytypecheck == ARMY_TYPE.ARMY_ARCHER) then
			ac = 1;	
		elseif (armytypecheck == ARMY_TYPE.ARMY_SPEARMAN) then
			spear = 1;
		elseif (armytypecheck == ARMY_TYPE.ARMY_CAVALRY) then
			cavalry = 1;
		elseif (armytypecheck == ARMY_TYPE.ARMY_WISER) then
			wiser = 1;
		elseif (armytypecheck == ARMY_TYPE.ARMY_SHIELDMAN) then
			shield = 1;
		end
	end

	if (ac + spear+ cavalry + wiser + shield > 3) then
		LuaDataAgt:AddBuff(13015, LuaDataAgt, nil);
	else 
		LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
	end
	
	LuaDataAgt:SetSkillEnable(false);
end
