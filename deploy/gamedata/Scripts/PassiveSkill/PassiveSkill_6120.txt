--技能生效,数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(true);
		return;
	end
end


local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT then
	    local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource then
			local j = math.random(1,10000);
			local Chance = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) * 100;
			if j < Chance then
				if not IsOnServer then
					local target = FilterValue:GetTarget();
					local TarPos = LuaDataAgt:GetCurPos();
					--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/CharEff_BaoJi", 0 , TarPos)
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_BaoJi", LuaDataAgt, TarPos, TarPos, 0, true);
				end
				iHp = 2*iHp;
			end
			FilterValue.m_fFoo1 = iHp;
		end
	end

	return FilterValue;
end

	return 0;
