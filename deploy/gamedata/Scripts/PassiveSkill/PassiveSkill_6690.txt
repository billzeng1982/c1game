local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	LuaDataAgt:SetSkillEnable(false);
end

--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	local iHp = 0 ;
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT then
		if IsSource and not IsOnServer then
			--[[
			local SpearRatio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId,1);
			print("SpearRatio="..SpearRatio);
			local target = FilterValue:GetTarget();
			local sex = target:GetGeneralSex();
			if sex == 0 then 
				--男性伤害增加
				SpearRatio = SpearRatio + 10;
			end
			iHp = FilterValue.m_fFoo1;
			print("damge=="..iHp);
			iHp = iHp *(1+SpearRatio/100);
			FilterValue.m_fFoo1 = iHp ;
			--]]
			local StunRatio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId,1);
			local target = FilterValue:GetTarget();
			local sex = target:GetGeneralSex();
			if sex == 0 then 
				--男性概率+10
				StunRatio = StunRatio + 10;
			end
			
			math.randomseed(os.time());
			local randomValue = math.random(1,100);

			if randomValue <= StunRatio then
				target:AddBuff(ThisScriptId, LuaDataAgt, nil);
			end
		end
	end
	print("damge=="..iHp);
	return FilterValue;
end
