--技能生效,数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	LuaDataAgt:SetSkillEnable(false);
end

local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SPEAR then
	    local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource then
			local Chance = LuaDataAgt: GetPassiveSkillBaseValue(ThisScriptId, 1) * 100;
			local j = math.random(1,10000);
			if j < Chance then
				local criticaldmg = 150;
				FilterValue.m_fFoo1 = iHp * criticaldmg / 100;

				if not IsOnServer then
					local TarPos = LuaDataAgt:GetCurPos();
					--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/CharEff_BaoJi", 0, TarPos);
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_BaoJi", LuaDataAgt, TarPos, TarPos, 0, true);
				end

				--local AddBlood = -FilterValue.m_fFoo1 * 0.1
				--LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, AddBlood, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_COMM, ThisScriptId, 0);
				--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/3D_CharEff_HuiFu", 1,  TarPos);
			end
		end
	end
	
	return FilterValue;
    
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local CriticalDmgRatio = rSource:GetActiveSkillBaseValue(0);
	--local CriticalDmgRatio = 100000;
	
	print ("CriticalDmgRatio="..CriticalDmgRatio);
	iRet = target:GetHp(VALUE_TYPE.RUNTIME_CONF) * (1+CriticalDmgRatio/100);
	return -iRet
end

PassiveSkill_6400 = {

	func_doFormula = doFormula;
