local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SPEAR then
	    local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource  then
			local target = FilterValue:GetTarget();
			local maxHp = target:GetHp(VALUE_TYPE.RUNTIME_CONF);
			local curHp = target:GetHp(VALUE_TYPE.CURRENT);
			if curHp < 0.45 * maxHp then
				local Extradmg = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1);
				FilterValue.m_fFoo1 = iHp + Extradmg;

				if not IsOnServer then
					--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_EWaiShangHai", 1 , LuaDataAgt:GetCurPos());
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_EWaiShangHai", LuaDataAgt, LuaDataAgt:GetCurPos(), LuaDataAgt:GetCurPos(), 1.5, true);
				end
			end
		end
	end

	return FilterValue;
end
