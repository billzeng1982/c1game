--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 一场战斗中只生效一次 然后不会Update
	LuaDataAgt:SetSkillEnable(false);
end

--技能过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
			
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_NORMAL then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource then --判断伤害是不是自己造成的
			local CurrHp = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT);
			local Ratio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId,1) / 100;
			local AddDMG = CurrHp * Ratio;
			
			FilterValue.m_fFoo1 = iHp - AddDMG;  --对伤害进行处理
		end
	end
	
	return FilterValue;
end

end
