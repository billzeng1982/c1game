--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--被动计找目标
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	--全军加属性 全军加的是一样的 
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF); --找己方TARGET_GROUP.SELF
	for i = 0,targetList.Count-1 do
		-- 通过添加Buff来达到属性同步
		targetList:get_Item(i):AddBuff(ThisScriptId, LuaDataAgt, nil);
	end
	
	-- 被动技能一场比赛只生效一次
	LuaDataAgt:SetSkillEnable(false);
end
