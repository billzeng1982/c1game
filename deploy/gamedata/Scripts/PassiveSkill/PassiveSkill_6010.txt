--[[
名称：弓腰姬
类型：被动技能
描述：弓箭伤害会溅射到附近的敌方单位身上
--]]
local ThisScriptId = 6010; 

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 添加血量过滤处理
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_PASSIVESKILL, ThisScriptId, 0);
	
	print("RunBorn PassiveSkill Id = "..ThisScriptId);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
end

--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 一场战斗中只生效一次 然后不会Update
	LuaDataAgt:SetSkillEnable(false);
end

--技能公式
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = iDamageRef;
	local DMGRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	iRet = iRet * DMGRatio;
	return iRet;
end

--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource and not IsOnServer then
			-- 在敌人周围找对象
			local target = FilterValue:GetTarget();
			
			target:ClearTarget();
			local targetList = target:FindTargetEx(TARGET_GROUP.SELF, SHAPE_TYPE.CIRCLE, 0, 0, 16, 0);
			
			local cnt = targetList.Count;
			for i=0,cnt-1 do
				local rTarget = targetList:get_Item(i);
				if not rTarget:Equal(target) and not rTarget:HasBuff(5001) then
					rTarget:ChgHp(VALUE_TYPE.CURRENT, iHp, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_PASSIVESKILL, ThisScriptId, 0);
				end
			end
		end
	end
	
	return FilterValue;
end

--这个ID要和脚本ID保持一致
PassiveSkill_6010 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
