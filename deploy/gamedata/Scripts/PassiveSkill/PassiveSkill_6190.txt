local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT then
	    local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource then
			local target = FilterValue:GetTarget();
			local maxHp = target:GetHp(VALUE_TYPE.RUNTIME_CONF);
			local curHp = target:GetHp(VALUE_TYPE.CURRENT);
			if curHp < 0.5 * maxHp then
				local Ratio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
				FilterValue.m_fFoo1 = iHp * (1 + Ratio);
				
				if not IsOnServer then
					local TarPos = target:GetCurPos();
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/UI_CharEff_EWaiShangHai", LuaDataAgt, TarPos, TarPos, 0, true);
					--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/CharEff_BaoJi", 1 , TarPos);
					LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_BaoJi", LuaDataAgt, TarPos, TarPos, 1, true);
				end
			end
		end
	end

	return FilterValue;
end
