--[[
名称：韬光养晦
类型：被动技能
描述：死亡时全军复活时间-10
--]]
local ThisScriptId = 5551;

--技能诞生处理
local function runBorn(LuaDataAgt)
	-- 添加血量过滤处理
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	--找之前先清一次
	LuaDataAgt:ClearTarget();
    local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	LuaDataAgt:RemoveTroopNotDead(targetList);
end


--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	-- 一场战斗中只生效一次
	LuaDataAgt:SetSkillEnable(false);
end

--Skill过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	local count = targetList.Count; 
	if count <= 0 then
		return FilterValue;
	end
	if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if not IsOnServer and not IsSource and iHp < 0 then
			local iCurHp = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT);
			if iCurHp + iHp <= 0 then
	            for i = 0,count - 1 do
		           targetList:get_Item(i):ChgRevivalTime(VALUE_TYPE.CURRENT,-10 );
	            end
			end
		end
	end
	
	return FilterValue;
end

PassiveSkill_5551 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	
	func_doFilter = doFilter;
}
