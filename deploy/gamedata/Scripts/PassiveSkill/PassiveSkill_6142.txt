--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	local Chgvalue = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1);
	
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.ENEMY);
	
	for i = 0,targetList.Count - 1 do
		targetList:get_Item(i):ChgStrDef(VALUE_TYPE.CURRENT,-Chgvalue);
	end
	
	LuaDataAgt:SetSkillEnable(false);
end
