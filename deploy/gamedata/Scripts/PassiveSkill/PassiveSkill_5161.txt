--技能生效，数值处理
local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	
	local iGroup = LuaDataAgt:GetGoGroup();
	local currPos = LuaDataAgt:GetCurPos();
	local prevPosZ = LuaDataAgt:GetShareValue(0);
	
	if iGroup == 1 then
		if (currPos.z > prevPosZ) then
			LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
		else
			LuaDataAgt:DelBuff(ThisScriptId, BUFF_TYPE.BUFF_TYPE_NONE, LuaDataAgt);
		end
	else
		if (currPos.z < prevPosZ) then
			LuaDataAgt:AddBuff(ThisScriptId, LuaDataAgt, nil);
		else
			LuaDataAgt:DelBuff(ThisScriptId, BUFF_TYPE.BUFF_TYPE_NONE, LuaDataAgt);
		end
	end
	
	
	LuaDataAgt:SetShareValue(0, currPos.z);
end
