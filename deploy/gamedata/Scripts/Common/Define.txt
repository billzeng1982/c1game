local __print = print;
if IsDebug then
	function print(...)
		-- get the point of call this print, the getinfo Level is 2(0 is getinfo self)
		-- 'S': fills in the fields source, linedefined, lastlinedefined, what, and short_src;
		local info = debug.getinfo(2, 'S')
		
		local filename = info.source;
		if string.find(filename, "/") then
			filename = string.match(filename, ".+/([^/]*)$")
		end
		
		__print(">>>From " .. filename .. " : ", ...)
	end
else
	function print(...)
		-- do nothing
	end
end
