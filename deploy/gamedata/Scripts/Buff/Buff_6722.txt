local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if IsSource and iHp < 0 then
			local Outdmg = 0.10;
			FilterValue.m_fFoo1 = iHp * (1 - Outdmg)

			if not IsOnServer then
				local TarPos = LuaDataAgt:GetCurPos();
				--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_JianShang", 4 , TarPos);
				LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JianShang", LuaDataAgt, TarPos, TarPos, 4, true);
			end
		end
	end
	
	return FilterValue;
end
