--Buff诞生处理
local function runBorn(LuaDataAgt)
	--buff时间设置
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 4);
	
	--过滤攻城打断条件
	LuaDataAgt:SetFilter(FILTER_TYPE.COND, TARGET_GROUP.NONE, 0);
	
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
end

	return true;

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.COND then
		if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_ATK_CITYPAUSE then
			-- 不被白兵暂停攻城，不被弓箭兵射击减速
			FilterValue.m_fFoo1 = 0;
		end
	end
	
	--攻城伤害增加
    if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;	
		if IsSource then
			if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_CITYWALL or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_CITYGATE then
			    local extra = 15;
				FilterValue.m_fFoo1 = iHp * (1 + extra/100);
			end
		end	
	end
	
	return FilterValue;
end


Buff_16 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
	func_doFilter = doFilter;
}
