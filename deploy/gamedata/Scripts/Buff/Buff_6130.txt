local ThisScriptId = 6130;

local function runBorn(LuaDataAgt)
	--属性加减
	local DodgeChance = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) * 10; --闪避1000为100%
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_DODGE, DodgeChance);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_ShanBiShangSheng", LuaDataAgt, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end
end
