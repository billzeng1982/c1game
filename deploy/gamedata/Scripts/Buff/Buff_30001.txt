local ThisScriptId = 30001;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	Source = LuaDataAgt:GetBuffSource();
	
	--buff时间设置
	local itime = 10;
	
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then     --秘籍1：增加时长
	    itime = itime * 1.1;
	end
	
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	--属性加减
	local Base = LuaDataAgt:GetActiveSkillBaseValue(2);
	local Ratio = LuaDataAgt:GetActiveSkillRateValue(2);
	local defChg = -5000;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, defChg);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end 
end
