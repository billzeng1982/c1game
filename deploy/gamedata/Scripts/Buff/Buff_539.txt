local ThisScriptId = 539;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	--buff时间设置
	local itime = LuaDataAgt:GetActiveSkillAdditionValue(2);
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then     --秘籍1：增加时长
	    itime = itime * 1.1;
	end
	
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	--属性加减
	local StrAtkChg = LuaDataAgt:GetActiveSkillBaseValue(1);
	local StrDefChg = LuaDataAgt:GetActiveSkillBaseValue(4);
	local WitDefChg = LuaDataAgt:GetActiveSkillBaseValue(5);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, StrAtkChg);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, StrDefChg);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, WitDefChg);
	if LuaDataAgt:IsHaveActiveSkillCheats(2) then     --秘籍2：加法防
		LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, WitDefChg * 0.5);
	end
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

	local StrDefChg = LuaDataAgt:GetActiveSkillBaseValue(4);

	end

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
