--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	local rSource = FilterValue:GetSource();
	if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and not IsSource and not IsOnServer then
			local Source = FilterValue:GetSource();
			local Ratio = 0.3；
			local iHpback = iHp * Ratio;
			local MaxDamge = rSource:GetStrAtk(VALUE_TYPE.CURRENT);
			MaxDamge = MaxDamge * 5;
			iHpback = math.max(-MaxDamge,iHpback);
			Source:ChgHp(VALUE_TYPE.CURRENT, iHpback, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_COMM, ThisScriptId, 0);
		end
	end
	
	return FilterValue;
end
