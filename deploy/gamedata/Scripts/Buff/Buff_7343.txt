--Buff诞生处理
local function runBorn(LuaDataAgt)
	--buff时间设置
	local source = LuaDataAgt:GetBuffSource();
	
	local itime =source:GetActiveSkillAdditionValue(3);								--BUFF时间参数
	local speed =source:GetActiveSkillAdditionValue(2);								--速度参数
	if source:IsHaveActiveSkillCheats(2) then     --秘籍2：增加减速效果
	    speed = speed * 1.2;
	end
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);						    --BUFF时间
	print("itime="..itime);
	print("speed="..speed);
	LuaDataAgt:ChgSpeedByRatio( 1 - speed / 100);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
