local ThisScriptId = 6140; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	print("runBorn=="..ThisScriptId);
	local rsource = LuaDataAgt:GetBuffSource();
	
	
	--属性加减
	local buffLevel = LuaDataAgt:GetBuffLevel();	--实现叠加机制
	if buffLevel >= 3 then 
		buffLevel = 3;
	end
	local Ratio = LuaDataAgt: GetPassiveSkillBaseValue(ThisScriptId, 1)/100;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF);
	AddStrDef = AddStrDef * Ratio * buffLevel;
	
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddStrDef);
	local iDef = LuaDataAgt:GetStrDef(VALUE_TYPE.CURRENT) ;
	print("iDef1=="..iDef);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
