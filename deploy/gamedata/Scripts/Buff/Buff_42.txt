local ThisScriptId = 42;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	local ap = source:GetWitAtk(VALUE_TYPE.CURRENT);
	local def = LuaDataAgt:GetWitDef(VALUE_TYPE.CURRENT);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 5 + ap / def);
	
	-- 设置公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
end
