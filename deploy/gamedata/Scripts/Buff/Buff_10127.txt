local ThisScriptId = 10127;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();

	--buff时间设置
	local itime =source:GetActiveSkillAdditionValue(4);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);	--BUFF时间
	
	--属性加减
	local atk = source:GetActiveSkillAdditionValue(1);
	local def = source:GetActiveSkillAdditionValue(2);	--加防的参数
	local speed =source:GetActiveSkillAdditionValue(3);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, atk);	--加攻
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, def);	--加防
	LuaDataAgt:ChgSpeedByRatio(1 + speed / 100);	--加速
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff驱散条件判断

--Buff驱散生效
local function runDead(LuaDataAgt)
    local source = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local atk = source:GetActiveSkillAdditionValue(1);
	local def = source:GetActiveSkillAdditionValue(2);	--加防的参数
	local speed =source:GetActiveSkillAdditionValue(3);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -atk);	--加攻
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -def);	--加防
	LuaDataAgt:ChgSpeedByRatio(-1 - speed / 100);	--加速
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
