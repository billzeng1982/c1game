--[[
名称：以少敌多
类型：被动技能
描述：于禁提升友方弓兵的法防，并提提升栅栏血量
--]]
local ThisScriptId = 6802;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_FaFangShangSheng", 4.5 , TarPos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangShangSheng", LuaDataAgt, TarPos, TarPos, 4.5, true);
	end
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

	return true;

end
