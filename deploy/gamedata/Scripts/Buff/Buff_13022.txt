local ThisScriptId = 13022; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local itime = source: GetPassiveSkillBaseValue(6700, 2)
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	local Speed = source:GetPassiveSkillBaseValue(6700,1) / 100;
	LuaDataAgt:ChgSpeedByRatio(1 + Speed);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local CurPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JiaSu", LuaDataAgt, CurPos, CurPos, 1, true);
	end
end
