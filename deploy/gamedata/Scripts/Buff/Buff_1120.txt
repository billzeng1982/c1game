local ThisScriptId = 1120;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local curLevel = LuaDataAgt:GetMSLevel4Buff();  	    -- 获得等级
	local curProgress = LuaDataAgt:GetMSProgress4Buff();	-- 获得进度
	local curAddRatio = LuaDataAgt:GetMSAddRatioPower4Buff();	-- 获得效果增益
	local maxValue = 20;
	local minValue = 5;
	
	--计算公式
	local addValue = 0;
	if curProgress >= 1.0 then
		addValue = maxValue;
	else
		addValue = minValue + curProgress * (maxValue - minValue) * 0.6;
	end
	
	if not IsOnServer then
		--增加复活时间
		LuaDataAgt:ChgRevivalTime(VALUE_TYPE.ORIGIN_CONF, addValue * curAddRatio);
	end
end
