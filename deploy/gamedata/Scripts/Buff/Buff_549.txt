local function takeEffect(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	local dancePower = source:GetActiveSkillShareValue(0);
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, dancePower, source, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 10, 0);
end
