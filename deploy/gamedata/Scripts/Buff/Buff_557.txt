--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--buff时间设置
	local itime = source:GetActiveSkillBaseValue(3);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		local OffSet = Vector3(35, 10, 0);
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_XuanYun", 0.5, OffSet);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_XuanYun", LuaDataAgt, OffSet, OffSet, 0.5, true);
	end
	
	--设置过滤器
	LuaDataAgt:SetFilter(FILTER_TYPE.COND, TARGET_GROUP.NONE, 0);
	LuaDataAgt:SetFilter(FILTER_TYPE.POS, TARGET_GROUP.NONE, -2);
end

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.COND then
		-- 不能单挑，被单挑
		if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_SOLO then
			FilterValue.m_fFoo1 = 0;
		end
		
		if IsSource then
			--不能使能兵种技能，不能释放武将技能
			if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_ARMYSKILL or FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_GENERALSKILL then
				FilterValue.m_fFoo1 = 0;
			end
			
			--不能攻城 
			if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_ATK_CITY or FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_ATK_OBSTACLE then
				FilterValue.m_fFoo1 = 0;
			end
			
			-- 不能进行普通攻击
			if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_ATK_NORMAL then
				FilterValue.m_fFoo1 = 0;
			end
		end
	end
	
	if iFilterType == FILTER_TYPE.POS then
		--不能变位置
		local destPos = LuaDataAgt:GetCurPos();
		FilterValue.m_fFoo1 = destPos.x;
		FilterValue.m_fFoo2 = destPos.y;
		FilterValue.m_fFoo3 = destPos.z;

		--print("destPos--"..destPos.x.."--"..destPos.y.."--"..destPos.z);
	end
	
	return FilterValue;
end

	func_doFilter = doFilter;
