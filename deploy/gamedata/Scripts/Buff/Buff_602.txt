--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	--伏兵状态下释放技能1.5倍伤害
	if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_GENERALSKILL and IsSource then
		if FilterValue.m_fFoo1 < 0 then
			local iHp = FilterValue.m_fFoo1;
			FilterValue.m_fFoo1 = iHp * 1.1;
			local iTime = LuaDataAgt:GetShareTimer(1); 
			if not IsOnServer and iTime < 0 then
				local OffSet = LuaDataAgt:GetCurPos();
				LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_ShangHaiQiangHua", LuaDataAgt, OffSet, OffSet, 0, true); -- 延迟0播该特效
				LuaDataAgt:SetShareTimer(1,8);
			end
		end
	end
	
	return FilterValue;
end
