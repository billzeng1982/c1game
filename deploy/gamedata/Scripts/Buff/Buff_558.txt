local ThisScriptId = 558;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();

	local itime = source:GetActiveSkillBaseValue(3);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, -5);
	--减双防
	LuaDataAgt:SetShareValue(1,0);
	LuaDataAgt:SetShareValue(2,0);
	local iSDef = LuaDataAgt:GetStrDef(VALUE_TYPE.RUNTIME_CONF)*0.3;
	local iWDef = LuaDataAgt:GetWitDef(VALUE_TYPE.RUNTIME_CONF)*0.3;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT,-iSDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT,-iWDef);
	LuaDataAgt:SetShareValue(1,iSDef);
	LuaDataAgt:SetShareValue(2,iWDef);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and not IsSource and not IsOnServer then
			if (FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SPEAR or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_NORMAL or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_RUSH or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_FACE or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_AMBUSH or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_GENERALSKILL) then

				local curHp = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT);

				if (curHp + iHp) <= 0 then
					local rSource = LuaDataAgt:GetBuffSource();
					local addhp = rSource:GetHp(VALUE_TYPE.RUNTIME_CONF);
					local huifuratio = rSource:GetActiveSkillBaseValue(6);
					addhp = addhp * huifuratio;

					local curpos = rSource:GetCurPos();
					--rSource:PlayFX("Prefabs/BasicEffect/3D_CharEff_HuiFu", rSource, curpos, curpos, 0.5, true);

					rSource:ChgHp(VALUE_TYPE.CURRENT, addhp, rSource,VALUE_CHG_TYPE.CHG_HP_COMM, 0, 221);
				end
			end
		end
	end

	return FilterValue;
end


Buff_558 = {

	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
