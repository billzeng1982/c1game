local ThisScriptId = 10116;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	local itime = source:GetActiveSkillAdditionValue(2);		--BUFF时间
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	if not IsOnServer then
		-- 播放特效（带延迟）
		local currPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BuffEffect/Buff_JiaXue", LuaDataAgt, currPos, currPos, 0.5, false);
	end
	
	LuaDataAgt:SetFilter(FILTER_TYPE.TRIG, TARGET_GROUP.NONE, 0);
end

--Buff生效
local function takeEffect(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, 0, source, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 10115, 0);				--调用的10115号的技能公式
	
end

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.TRIG and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_TRIG_BUFF_ADD_END then
		--受到敌人的技能控制
		if not IsOnServer and not IsSource then
			local iBuffType = BUFF_TYPE.IntToEnum(FilterValue.m_fFoo2);
			if (iBuffType == BUFF_TYPE.BUFF_TYPE_SPEED_LOSS or
				iBuffType == BUFF_TYPE.BUFF_TYPE_FAINT or 
				iBuffType == BUFF_TYPE.BUFF_TYPE_RESTRAIN or
				iBuffType == BUFF_TYPE.BUFF_TYPE_SCOFF) then
				
				FilterValue.m_fFoo1 = 0;
			end
		end
	end
	
	return FilterValue;
end

Buff_10116 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
	
	func_doFilter = doFilter;
}
