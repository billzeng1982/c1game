local ThisScriptId = 1200;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 10);			--BUFF时间
	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, 50);   --加攻
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, 50);	 --加防
	LuaDataAgt:ChgSpeedByRatio(1 + 50/100);		 --加速
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff驱散条件判断

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -50);	 
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -50);
	--移除加速
	LuaDataAgt:ChgSpeedByRatio(-1 - 50/100);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end 
end

end
