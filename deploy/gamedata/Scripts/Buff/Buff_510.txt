local ThisScriptId = 510;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	--buff时间设置
	local itime = LuaDataAgt:GetActiveSkillAdditionValue(3);
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then     --秘籍1：增加时长
	    itime = itime * 1.1;
	end

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	--属性加减
	local WitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.CURRENT);
	local Base = LuaDataAgt:GetActiveSkillBaseValue(2);
	local Ratio = LuaDataAgt:GetActiveSkillRateValue(2);
	local defChg = Base+ ((WitAtk - 480) * Ratio) + 240;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, defChg);
	
	if LuaDataAgt:IsHaveActiveSkillCheats(2) then     --秘籍2：加法防
	    LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, defChg * 0.5);
	end
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end 
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)
	
	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local WitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.CURRENT);
	local Base = LuaDataAgt:GetActiveSkillBaseValue(2);
	local Ratio = LuaDataAgt:GetActiveSkillRateValue(2);
	local defChg = Base+ ((WitAtk - 480) * Ratio) + 240;
	
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -defChg);
	
	if LuaDataAgt:IsHaveActiveSkillCheats(2) then     --秘籍2：加法防
	    LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -defChg * 0.5);
	end
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end 
end

	return true;
end

end
					
Buff_510 = {

	func_runDead = runDead;
