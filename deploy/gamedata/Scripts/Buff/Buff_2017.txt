--[[
名称：觉醒奇门遁甲
类型：武将技能Buff
描述：减血眩晕几率增加
--]]
local ThisScriptId = 2017;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	-- Buff时间设置
	local itime = source:GetActiveSkillAdditionValue(3);
	if source:IsHaveActiveSkillCheats(2) then     --秘籍2：时间增长
	    itime = itime * 1.1;
	end
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	-- 设置Buff公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)
	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)

end

--Buff生效条件判断
local function checkEffectCond(LuaDataAgt)
	return true;
end

--Buff生效
local function takeEffect(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, 1, source, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, ThisScriptId, 0);

	local k = LuaDataAgt:GetStrAtk(VALUE_TYPE.CURRENT);
	math.randomseed(tostring(os.time() + k):reverse():sub(1, 6)) ;
	local j = math.random(1,100);
	if j < 41 then
		--武器觉醒眩晕概率上升到40%
		LuaDataAgt:AddBuff(555, LuaDataAgt, nil);
		
		--眩晕buff
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_XuanYun", LuaDataAgt, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end
end

local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)

end

--Buff公式处理
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	local WitAtk = rSource:GetWitAtk(VALUE_TYPE.CURRENT);
	local WitDef = rTarget:GetWitDef(VALUE_TYPE.CURRENT);
	local Base = rSource:GetActiveSkillBaseValue(2);
	local MinDmg = 0.2 * Base;
	local Ratio = rSource:GetActiveSkillRateValue(2);
	
	iRet = Base + (WitAtk - WitDef) * Ratio;
	iRet = math.max(iRet, MinDmg);
	
	return -iRet;
end

Buff_2017 = {
func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
}
