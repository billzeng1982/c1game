local ThisScriptId = 6132;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, -2);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 999);
	
	--属性加减
	local Ratio = rSource:GetPassiveSkillBaseValue(6132, 1) / 100;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * Ratio;		
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * Ratio;	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, AddStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, AddWitAtk);	
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_QuanShuXingSS", 3 , TarPos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_QuanShuXingSS", LuaDataAgt, TarPos, TarPos, 3, true);
	end
end

--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	local iHp = FilterValue.m_fFoo1;
	if iFilterType == FILTER_TYPE.HP and iHp < 0 and not IsSource and not IsOnServer and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_GENERALSKILL then
		--[[
		local i = 1500 ;
		math.randomseed(os.time()) ;
		local j = math.random(1,10000);
		if i > j then 
			iHp = 0 ;
			local TarPos = LuaDataAgt:GetCurPos();
			LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JianShang", LuaDataAgt, TarPos, TarPos, 3, true);
		end
		]]
		iHp = iHp * 0.8 ;--减伤20%
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JianShang", LuaDataAgt, TarPos, TarPos, 0, true);
		FilterValue.m_fFoo1 = iHp ;
		LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.CURRENT, 0);
	end

	return FilterValue;
end
