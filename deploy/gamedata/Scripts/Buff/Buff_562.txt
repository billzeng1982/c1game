--Buff生效
local function takeEffect(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	--dancepower*10，尾数0表示是打人，尾数1表示打城墙
	local dancePowerTran = source:GetActiveSkillShareValue(0) * 10;
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, dancePowerTran, source, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 2048, 0);
end

	func_takeEffect = takeEffect;
