--[[
名称:黄忠1
类型:被动技能
描述:黄忠每次射击都会增加暴击率，可以叠加
--]]
local ThisScriptId = 6320; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local buffLevel = LuaDataAgt:GetBuffLevel();	--实现叠加机制
	if buffLevel > 10 then
		buffLevel = 10; --设置叠加上限
	end
	
	local ChgValue = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) * buffLevel * 10;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_CRITICAL, ChgValue);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff驱散条件判断

local function runDead(LuaDataAgt)
	local buffLevel = LuaDataAgt:GetBuffLevel();	--实现叠加机制
	if buffLevel > 10 then
		buffLevel = 10; --设置叠加上限
	end
	
	local ChgValue = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) * buffLevel * 10;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_CRITICAL, -ChgValue);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
