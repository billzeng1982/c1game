local ThisScriptId = 595;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	local itime = source:GetActiveSkillAdditionValue(3);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_CRITICAL, 400);
	
	if LuaDataAgt:IsHaveActiveSkillCheats(1) then  --秘籍1： 加速50%
	    LuaDataAgt: ChgSpeedByRatio(1.5);
	end
end
