local ThisScriptId = 99;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	if not IsOnServer then
		-- 播放特效（带延迟）
		local currPos = LuaDataAgt:GetCurPos();
		local destPos = LuaDataAgt:GetTroopOppositeWallPos();
		destPos.x = currPos.x;
		destPos.y = currPos.y;
		LuaDataAgt:PlayFX("Prefabs/SkillEffect/Skill_Jgss_Bullet", LuaDataAgt, currPos, destPos, 0.5, false); -- 延迟0.5播该特效
	end
end

--Buff生效
local function takeEffect(LuaDataAgt)
	local currPos = LuaDataAgt:GetCurPos();
	local destPos = LuaDataAgt:GetTroopOppositeWallPos();
	destPos.x = currPos.x;
	destPos.y = currPos.y;
	LuaDataAgt:PlayFX("Prefabs/SkillEffect/Skill_Jgss_End", LuaDataAgt, destPos, destPos, 0, false); --子弹爆炸
	
	local dancePower = LuaDataAgt:GetActiveSkillShareValue(0);
	
	local city = LuaDataAgt:FindTargetCity(TARGET_GROUP.ENEMY);
	city:ChgHp(VALUE_TYPE.CURRENT, dancePower, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 9, 0);
end

	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
