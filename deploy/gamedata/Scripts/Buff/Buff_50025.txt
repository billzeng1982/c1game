local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if iFilterType == FILTER_TYPE.HP then
	    local iHp = FilterValue.m_fFoo1;

		if iHp < 0 and IsSource and not IsOnServer then
		--print("shanghai1--------.....",iHp);
		local MaxHp = LuaDataAgt:GetHp(VALUE_TYPE.RUNTIME_CONF);
		local CurHp = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT) * 10;
		--print("MaxHp---------",MaxHp);
		--print("CurHp---------",CurHp);
		local times = 10 - math.ceil(CurHp/MaxHp);
		--print("CurHp/MaxHp---------:", math.ceil(CurHp/MaxHp));
		--print("times---------:", times);
		local ratio = LuaDataAgt:GetResTacticsBuffValue(ThisScriptId) * times;
		--print("ratio---------:", ratio);
		FilterValue.m_fFoo1 = iHp + iHp * ratio;
		--print("shanghai2---------....",FilterValue.m_fFoo1);
		end
	end
	
	return FilterValue;
end

	func_checkEffectCond = checkEffectCond;

	func_doFilter = doFilter;
