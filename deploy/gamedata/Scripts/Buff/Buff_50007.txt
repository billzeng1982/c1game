--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_fFoo1 < 0 then
		if not IsSource and not IsOnServer then
			local Chance = LuaDataAgt:GetResTacticsBuffValue(ThisScriptId) * 10000;
			--local Chance = 10000;
			local j = math.random(1, 10000);
			if j < Chance then
				local target = FilterValue:GetSource();
				target:AddBuff(50008, LuaDataAgt, nil);
			end
		end
	end
	
	return FilterValue;
end

	func_takeEffect = takeEffect;
