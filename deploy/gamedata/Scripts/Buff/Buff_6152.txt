描述:刘备减少蜀国武将所受的伤害 如果蜀国武将大于等于4个效果翻倍<color=#53de1c>20</color>秒

local ThisScriptId = 6152;


--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rsource = LuaDataAgt:GetBuffSource();

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 999);

	--属性加减
	local DodgeChance = rsource: GetPassiveSkillBaseValue(ThisScriptId,1) / 100;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_DODGE, DodgeChance);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_JianShang", 3 , TarPos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JianShang", LuaDataAgt, TarPos, TarPos, 3, true);
	end
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local rsource = LuaDataAgt:GetBuffSource();

	--属性加减
	local DodgeChance = rsource: GetPassiveSkillBaseValue(ThisScriptId,1) / 100;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_DODGE, -DodgeChance);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff生效条件判断

	return true;
end

end

Buff_6152 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
}
