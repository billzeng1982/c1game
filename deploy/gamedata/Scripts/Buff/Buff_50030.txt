--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource and not IsOnServer then
			if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT then
				local Chance = LuaDataAgt:GetResTacticsBuffValue(ThisScriptId) * 10000;
				--local Chance = 10000;
				local j = math.random(1, 10000);
				if j < Chance then
					local target = FilterValue:GetTarget();
					target:AddBuff(50031, LuaDataAgt, nil);
				end
			end
		end
	end
end
