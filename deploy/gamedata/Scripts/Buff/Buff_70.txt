local ThisScriptId = 70;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--buff时间设置
	local itime = 6;
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);

	--属性加减
	local AddAtk = source:GetActiveSkillBaseValue(2);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddAtk);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos =  LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongXiaJiang", LuaDataAgt, TarPos, TarPos, 0.5, true);
	end
end
