--Buff诞生处理
local function runBorn(LuaDataAgt)
	--属性加减
	  local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * 0.15;
	  local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * 0.15;		
	  local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * 0.15;
	  local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * 0.15;	
	  LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT,AddStrAtk);
	  LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT,AddWitAtk);	
	  LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT,AddStrDef);
	  LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT,AddWitDef);
	  LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 10);
	  print("B");

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
