--Buff诞生处理
local function runBorn(LuaDataAgt)

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 8);
	local curAddRatio = LuaDataAgt:GetMSAddValue4Buff() / 100;	-- 获得效果增益
	local iStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.RUNTIME_CONF);
	local iWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.RUNTIME_CONF);
	iStrAtk = iStrAtk * curAddRatio;
	print("curAddRatio=="..curAddRatio);
	iWitAtk = iWitAtk * curAddRatio;
	
	LuaDataAgt:SetShareValue(0,iStrAtk);
	LuaDataAgt:SetShareValue(1,iWitAtk);
	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT,iStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT,iWitAtk);
	--[[
	local curProgress = LuaDataAgt:GetMSProgress4Buff();	-- 获得进度
	local curAddRatio = LuaDataAgt:GetMSAddRatioPower4Buff();	-- 获得效果增益
	local maxValue1 = 600;
	local minValue1 = 150;	
	local maxValue2 = 700;
	local minValue2 = 150;
	
	--计算公式
	local addValue2 = 0;
	local addValue1 = 0;
	if curProgress >= 1.0 then
		addValue2 = maxValue2;
		addValue1 = maxValue1;
	else
		addValue2 = minValue2 + curProgress *( maxValue2 - minValue2) * 0.6;
		addValue1 = minValue1 + curProgress *( maxValue1 - minValue1) * 0.6;
	end
	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT,addValue1 * curAddRatio);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT,addValue2 * curAddRatio);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
	]]
end
