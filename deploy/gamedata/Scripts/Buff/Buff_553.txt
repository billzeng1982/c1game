local function runBorn(LuaDataAgt)
	--加暴击概率
	local CriticalChance = 400;
	--加暴击伤害
	--local CriticalRatio = 1000;
	
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_CRITICAL, CriticalChance);
	--LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_CRITICAL_VALUE, CriticalRatio);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local curpos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_BaoJiShangSheng",0.5,curpos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_BaoJiShangSheng", LuaDataAgt, curpos, curpos, 0.5, true);
	end
end
