名称：如虎而痴	
类型：被动技能Buff
描述：血量低于50%，加全属性20%，持续15秒，60秒内不重复触发

local ThisScriptId = 5524;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	--属性加减
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * 0.2;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * 0.2;		
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * 0.2;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * 0.2;	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT,AddStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT,AddWitAtk);	
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT,AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT,AddWitDef);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 15);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
