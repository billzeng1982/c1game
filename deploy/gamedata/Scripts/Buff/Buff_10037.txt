local ThisScriptId = 10037;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--buff时间设置
	local itime =source:GetActiveSkillAdditionValue(4);	--时间
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	--属性加减
	local atk = source:GetActiveSkillAdditionValue(3);   --物攻
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, atk);
	
	local speed = 1 + source:GetActiveSkillAdditionValue(2) / 100;	--速度
	LuaDataAgt:ChgSpeedByRatio(speed);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff驱散生效
local function runDead(LuaDataAgt)
    local source = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local atk = source:GetActiveSkillAdditionValue(3);   --物攻
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -atk);
	
	local speed = 1 + source:GetActiveSkillAdditionValue(2) / 100;	--速度
	LuaDataAgt:ChgSpeedByRatio(-speed);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end 
end
