local ThisScriptId = 13017;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();

	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(6730, 1) / 100;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_FaFangShangSheng", 2 , TarPos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangShangSheng", LuaDataAgt, TarPos, TarPos, 2, true);
	end
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();

	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(6730, 1) / 100;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
