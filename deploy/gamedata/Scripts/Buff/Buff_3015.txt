local ThisScriptId = 3015;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	-- 设置Buff公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
	-- 属性加减
	local originHp = LuaDataAgt:GetHp(VALUE_TYPE.ORIGIN_CONF);
	local addHp = 0.08 * originHp;
	
	-- 改变血量上限
	LuaDataAgt:ChgHp(VALUE_TYPE.RUNTIME_CONF, addHp, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
	
	if not IsOnServer then
		LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, addHp, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
		LuaDataAgt:RefreshFightNum();
	end
end
