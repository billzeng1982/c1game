--Buff生效
local function takeEffect(LuaDataAgt)
	LuaDataAgt:SetCurPos(LuaDataAgt:GetDeadPos(), true);
	
	-- 播放特效（带延迟）
	local currPos = LuaDataAgt:GetCurPos();
	LuaDataAgt:PlayFX("Prefabs/BuffEffect/Master_Fenyong", LuaDataAgt, currPos, currPos, 0.1, false);
	
	-- 生效一次，删除本Buff
	LuaDataAgt:DelBuff(ThisScriptId, BUFF_TYPE.BUFF_TYPE_NONE, LuaDataAgt);
end
