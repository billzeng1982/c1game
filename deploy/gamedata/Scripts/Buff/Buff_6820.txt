local ThisScriptId = 6820;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local Source = LuaDataAgt: GetBuffSource();
	
	--属性加减
	local Ratio = Source: GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangShangSheng", LuaDataAgt, TarPos, TarPos, 1, true); -- 延迟0播该特效
	end
end

--Buff驱散条件判断

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local Source = LuaDataAgt: GetBuffSource();
	
	--属性加减
	local Ratio = Source: GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
