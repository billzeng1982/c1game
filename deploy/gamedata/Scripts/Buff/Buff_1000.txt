--Buff诞生处理
local function runBorn(LuaDataAgt)
	local curLevel = LuaDataAgt:GetMSLevel4Buff();  	-- 获得等级
	local curProgress = LuaDataAgt:GetMSProgress4Buff();	-- 获得进度
	local curAddRatio = LuaDataAgt:GetMSAddRatioPower4Buff();	-- 获得效果增益
	local maxValue = 30;
	local minValue = 10;
	
	--计算公式
	local addValue = 0;
	if curProgress >= 1.0 then
		addValue = maxValue;
	else
		addValue = minValue + curProgress *( maxValue - minValue) * 0.6;
	end
	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT,addValue);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
