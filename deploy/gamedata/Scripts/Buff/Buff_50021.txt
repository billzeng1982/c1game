--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_fFoo1 < 0 then
		if not IsSource and not IsOnServer then
			LuaDataAgt:AddBuff(50022, LuaDataAgt, nil);
			
			local TarPos = LuaDataAgt:GetCurPos();
			LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JiaSu", target, TarPos, TarPos, 0, true); -- 延迟0播该特效
		end
	end
	
	return FilterValue;
end
