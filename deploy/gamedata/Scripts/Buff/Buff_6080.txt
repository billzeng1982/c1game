local ThisScriptId = 6080; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	--属性加减
	local buffLevel = LuaDataAgt:GetBuffLevel();	--实现叠加机制
	if buffLevel >= 3 then 
		buffLevel = 3;
	end
	
	local PassiveSkillRatio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddStrDef = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio * buffLevel;
	local iDef = LuaDataAgt:GetStrDef(VALUE_TYPE.CURRENT);
	print("iDef1=="..iDef);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddStrDef);
	iDef = LuaDataAgt:GetStrDef(VALUE_TYPE.CURRENT);
	print("iDef2=="..iDef);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

	end
end
