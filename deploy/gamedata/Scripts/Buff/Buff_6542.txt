--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	local GenderID = LuaDataAgt:GetGeneralSex();		
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and not IsSource then --判断伤害是不是自己造成的
			local source = LuaDataAgt:GetBuffSource();
			local Ratio = source:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	
			FilterValue.m_fFoo1 = iHp * (1-Ratio);  --对伤害进行处理
		end
	end
	
	return FilterValue;

end
