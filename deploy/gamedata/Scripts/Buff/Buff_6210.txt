--Buff诞生处理
local function runBorn(LuaDataAgt)
	local Source = LuaDataAgt:GetBuffSource();
	
	--属性加减
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, -2);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JianShang", LuaDataAgt, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end
end

--技能过滤
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and not IsSource and not IsOnServer then
		local iHp = FilterValue.m_fFoo1;
		local PassiveSkillRatio = LuaDataAgt:GetPassiveSkillBaseValue(6212,1);
		FilterValue.m_fFoo1 = iHp * (1 - PassiveSkillRatio / 100);
		
	end
	return FilterValue;
end
