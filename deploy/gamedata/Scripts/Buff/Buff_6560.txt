--[[
名称:刚直壮烈
类型:被动技能buff
描述:陈宫的射击会提升自身伤害
--]]
local ThisScriptId = 6560; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local Ratio = source:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, AddWitAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, AddWitAtk);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongShangSheng", LuaDataAgt, TarPos, TarPos, 1, true); -- 延迟0播该特效
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaGongShangSheng", LuaDataAgt, TarPos, TarPos, 2, true); -- 延迟1播该特效
	end
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)
	
	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local Ratio = source:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddWitAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, -AddWitAtk);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff生效条件判断
local function checkEffectCond(LuaDataAgt)
	return true;
end

--Buff生效
local function takeEffect(LuaDataAgt)
	
end

Buff_6560 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
}
