local ThisScriptId = 6492; 


--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
		--属性加减
	local buffLevel = LuaDataAgt:GetBuffLevel();	--实现叠加机制
	if buffLevel >= 5 then 
		buffLevel = 5;
	end
	
	
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio * buffLevel;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddStrDef);
	local iDef = LuaDataAgt:GetStrDef(VALUE_TYPE.CURRENT);
	print("iDef=="..iDef);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

	end
end
