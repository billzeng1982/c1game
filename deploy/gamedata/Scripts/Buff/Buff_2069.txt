local ThisScriptId = 2069;


--Buff诞生处理
local function runBorn(LuaDataAgt)
		
	--buff时间设置
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 6);
	
	--属性加减
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * 0.3;
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, AddWitAtk);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaGongShangSheng", LuaDataAgt, TarPos, TarPos, 1, true);
	end
end

--Buff驱散条件判断

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	--属性加减回退
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * 0.3;
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, -AddWitAtk);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

end
