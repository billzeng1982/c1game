local ThisScriptId = 6534;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	--测试
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 5);
	
	--属性加减
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * 0.25;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * 0.25;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddStrAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddStrDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
