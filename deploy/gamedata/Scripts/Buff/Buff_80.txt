local ThisScriptId = 80;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--buff时间设置
	local itime = source:GetActiveSkillBaseValue(2);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	--属性加减
	local Chgvalue = source:GetActiveSkillBaseValue(1)*10;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_CRITICAL, Chgvalue);
	
	
	if not IsOnServer then
		LuaDataAgt:ChgArcherRadiusByRatio(1.5);
		
		LuaDataAgt:RefreshFightNum();
	end 
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	
    local source = LuaDataAgt:GetBuffSource();
	
	--属性回退
	local Chgvalue = source:GetActiveSkillBaseValue(1)*10;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_CRITICAL, -Chgvalue);
	
	if not IsOnServer then
		LuaDataAgt:ChgArcherRadiusByRatio(-1.5);
		
		LuaDataAgt:RefreshFightNum();
	end 
end

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
--[[
	-- Filter所有者是否是发起对象
	
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.TRIG then
		--受到敌人的技能控制
		if not IsOnServer and not IsSource and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_TRIG_BUFF_ADD_END then
			local iBuffType = BUFF_TYPE.IntToEnum(FilterValue.m_fFoo2);
			if (iBuffType == BUFF_TYPE.BUFF_TYPE_SPEED_LOSS or
				iBuffType == BUFF_TYPE.BUFF_TYPE_FAINT or 
				iBuffType == BUFF_TYPE.BUFF_TYPE_RESTRAIN or
				iBuffType == BUFF_TYPE.BUFF_TYPE_SCOFF
				) then
				FilterValue.m_fFoo1 = 0;
			end
		end
	end
	
	return FilterValue;
--]]
end

	func_takeEffect = takeEffect;
	func_doFilter = doFilter;
