--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--属性加减,Ratio是调整的倍数
	local Ratio = 1;
	local StrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * (Ratio - 1);
	local WitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * (Ratio - 1);
	local StrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * (Ratio - 1);
	local WitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * (Ratio - 1);
	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, StrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, WitAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, StrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, WitDef);
	
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
end
