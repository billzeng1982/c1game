--[[
名称：直捣黄龙
类型：武将技能Buff
描述：提高攻城伤害，不被打断
--]]
local ThisScriptId = 10126;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();

	--buff时间设置
	local itime = source:GetActiveSkillAdditionValue(3);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);

	-- 设置Buff公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);

	--过滤攻城打断条件
	LuaDataAgt:SetFilter(FILTER_TYPE.COND, TARGET_GROUP.NONE, 0);

	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);
end

--Buff驱散条件判断

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end

end

	return true;
end

end

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());

	if iFilterType == FILTER_TYPE.COND then
		if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_ATK_CITYPAUSE then
			-- 不被白兵暂停攻城，不被弓箭兵射击减速
			FilterValue.m_fFoo1 = 0;
		end
	end

	--攻城伤害翻倍
    if iFilterType == FILTER_TYPE.HP then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 and IsSource then
			if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_CITYWALL or
				FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_CITYGATE then

				local source = LuaDataAgt:GetBuffSource();
				local extra = source:GetActiveSkillAdditionValue(1) / 100;
				FilterValue.m_fFoo1 = iHp * (1 + extra);

				if source:IsHaveActiveSkillCheats(1) then
					if not IsOnServer then
						local city = LuaDataAgt:FindTargetCity(TARGET_GROUP.SELF);
						city:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
					end
				end
			end
		end

		if iHp < 0 and not IsSource then
			-- 直捣黄龙减伤
			local source = LuaDataAgt:GetBuffSource();
			local reduction = source:GetActiveSkillAdditionValue(2) / 100;
			FilterValue.m_fFoo1 = iHp * (1 - reduction);
		end
	end

	return FilterValue;
end

--Buff公式处理
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	local iCheatsValue = rSource:GetActiveSkillCheatsValue(1);

	iRet = iCheatsValue;
	return iRet;
end

Buff_10126 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
	func_doFormula = doFormula;
	func_doFilter = doFilter;
}
