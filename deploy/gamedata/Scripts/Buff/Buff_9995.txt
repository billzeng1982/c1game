local ThisScriptId = 9995;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	
	--buff时间设置
	local itime = source:GetActiveSkillAdditionValue(2);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);
	
	--属性加减
	local act = source:GetActiveSkillAdditionValue(1);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, act);
	
	local speed = source:GetActiveSkillAdditionValue(0) / 100 + 1;
	LuaDataAgt:ChgSpeedByRatio(speed);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff驱散条件判断

	return false;

--Buff驱散生效
local function runDead(LuaDataAgt)
    local source = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local act = source:GetActiveSkillAdditionValue(1);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -act);
	
	local speed = source:GetActiveSkillAdditionValue(0) / 100 + 1;
	LuaDataAgt:ChgSpeedByRatio(-speed);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end 
end

end
