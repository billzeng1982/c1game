local ThisScriptId = 5184;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	--属性加减
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * 0.16;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * 0.16;		
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * 0.16;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * 0.16;	
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT,AddStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT,AddWitAtk);	
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT,AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT,AddWitDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
	
	--print("runBorn " .. ThisScriptId);
end
