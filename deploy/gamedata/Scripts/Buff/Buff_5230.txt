local ThisScriptId = 5230;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local buffLevel = LuaDataAgt:GetBuffLevel();
	--属性加减
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * 0.02;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddStrDef * buffLevel);
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
	--测试
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 10);
end
