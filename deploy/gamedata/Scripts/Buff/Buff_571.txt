--Buff生效
local function takeEffect(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	--local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.SELF, SHAPE_TYPE.CIRCLE, 0, 0, 50, 0);
	print("BuffCount =="..targetList.Count);
	for i = 0 , targetList.Count - 1 do
		local target = targetList:get_Item(i) ;
		if target:Equal(LuaDataAgt) then
		--不包括自己
		print("IsSelf!!!");
		else
			target:ChgHp(VALUE_TYPE.CURRENT, 1, rSource,VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 51, 0);
		end
	
	end 

end
