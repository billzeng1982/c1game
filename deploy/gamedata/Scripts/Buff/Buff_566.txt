local ThisScriptId = 566;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	print("successful");
	local Source = LuaDataAgt:GetBuffSource();

	--属性加减
	local iRatio = Source:GetActiveSkillBaseValue(2) / 100;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * iRatio;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * iRatio;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_WuFangShangSheng", 2 , TarPos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuFangShangSheng", LuaDataAgt, TarPos, TarPos, 0.5, true);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangShangSheng", LuaDataAgt, TarPos, TarPos, 1.5, true);
	end
end
