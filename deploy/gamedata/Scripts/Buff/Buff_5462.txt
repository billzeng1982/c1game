--Buff诞生处理
local function runBorn(LuaDataAgt)
	local buffLevel = LuaDataAgt:GetBuffLevel();
	LuaDataAgt:SetFilter(FILTER_TYPE.HP, TARGET_GROUP.NONE, 0);

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 10);
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * 0.05 * buffLevel;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT,-AddStrDef);
end
