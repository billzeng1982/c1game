local ThisScriptId = 6710; 
--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rsource = LuaDataAgt:GetBuffSource();
	
	local Ratio = rsource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddStrAtk);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongXiaJiang", LuaDataAgt, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end
end
