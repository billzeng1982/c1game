local ThisScriptId = 82;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rsource = LuaDataAgt:GetBuffSource();

	--buff时间设置
	local itime =  rsource:GetActiveSkillBaseValue(2);
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);

	local Chgvalue = rsource:GetActiveSkillBaseValue(3);
	--LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -Chgvalue);	--攻击下降
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -Chgvalue);	--防御下降
	--LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, -Chgvalue);	--攻击下降
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -Chgvalue);	--防御下降
	--LuaDataAgt:ChgSpeedByRatio(0.5);		 --速度减少

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt: PlayFX("Prefabs/BasicEffect/3D_CharEff_WuFangXiaJiang", LuaDataAgt, TarPos, TarPos,0.5, true);
		LuaDataAgt: PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangXiaJiang", LuaDataAgt, TarPos, TarPos,1, true);
	end
end
