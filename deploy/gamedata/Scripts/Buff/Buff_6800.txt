--[[
名称:于禁1
类型:被动技能
描述:于禁每次射击成功后增加自身物攻和武力10s<color=#53de1c>30%</color>
--]]
local ThisScriptId = 6800; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	--属性加减
	local Ratio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.RUNTIME_CONF) * Ratio;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.RUNTIME_CONF) * Ratio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, AddStrAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddStrDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongShangSheng", LuaDataAgt, TarPos, TarPos, 1, true); -- 延迟0播该特效
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuFangShangSheng", LuaDataAgt, TarPos, TarPos, 2, true); -- 延迟1播该特效
	end
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)
	
	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	--属性加减回退
	local Ratio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.RUNTIME_CONF) * Ratio;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.RUNTIME_CONF) * Ratio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddStrAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddStrDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff生效条件判断
local function checkEffectCond(LuaDataAgt)
	return true;
end

local function takeEffect(LuaDataAgt)

end

Buff_6800 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
}
