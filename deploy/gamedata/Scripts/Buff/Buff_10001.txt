--Buff生效
local function takeEffect(LuaDataAgt)
	if not IsOnServer then
		LuaDataAgt:ClearTarget();
		local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
		LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
		for i = 0 , targetList.Count-1 do
			local target = targetList:get_Item(i);
			local ChgValue = target:GetHp(VALUE_TYPE.RUNTIME_CONF) * 0.04;
			target:ChgHp(VALUE_TYPE.CURRENT, ChgValue, target, VALUE_CHG_TYPE.CHG_HP_COMM, 0, 221);
			local currPos = target:GetCurPos();
			target:PlayFX("Prefabs/BuffEffect/Buff_JiaXue", target, currPos, currPos, 0, true);
			target:PlayFX("Prefabs/BuffEffect/Master_Yuanjun", target, currPos, currPos, 0.3, true);
		end
	end

end
