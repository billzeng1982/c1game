--Buff诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 1);
	
	--设置过滤器
	LuaDataAgt:SetFilter(FILTER_TYPE.COND, TARGET_GROUP.NONE, 0);
	LuaDataAgt:SetFilter(FILTER_TYPE.POS, TARGET_GROUP.NONE, -2);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_ChenMo", LuaDataAgt, TarPos, TarPos, 0, true);
	end
end

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.COND and not IsOnServer then
		-- 不能单挑，被单挑
		if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_SOLO then
			FilterValue.m_fFoo1 = 0;
		end
		
		if IsSource then
			--不能释放武将技能
			if FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_COND_GENERALSKILL then
				FilterValue.m_fFoo1 = 0;
			end
		end
	end
	
	return FilterValue;
end
