local ThisScriptId = 570;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	print("ThisScriptId =="..ThisScriptId);
	--属性加减
	local buffLevel = LuaDataAgt:GetBuffLevel();	--实现叠加机制
	if buffLevel >= 3 then 
		buffLevel = 3;
	end
	
	local StrAtkChg = LuaDataAgt:GetStrAtk(VALUE_TYPE.RUNTIME_CONF)*0.05;
	StrAtkChg = StrAtkChg * buffLevel;
	LuaDataAgt:SetShareValue(3,StrAtkChg);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, StrAtkChg);
	LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_ChaoFeng", LuaDataAgt, Vector3.zero, Vector3.zero, 0, true); -- 延迟0播该特效
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
