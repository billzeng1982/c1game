--[[
名称：屏气凝神
类型：星级被动
描述：每次弓箭射击提升自身的弓箭伤害4%，持续10s，最多可叠加8层
--]]
local ThisScriptId = 5005;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local buffLevel = LuaDataAgt:GetBuffLevel();

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 12);
	local AddRatio = buffLevel * 0.125;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, AddStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, AddWitAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);
	
	if not IsOnServer then
		local OffSet = Vector3(35,10,0);
		LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_QuanShuXingXiaJiang",0,OffSet);
	end
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	--属性加减回退
	local buffLevel = LuaDataAgt:GetBuffLevel();

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 12);
	local AddRatio = buffLevel * 0.125;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * AddRatio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, -AddWitAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);
end

--Buff生效条件判断
local function checkEffectCond(LuaDataAgt)
	return true;
end

end
