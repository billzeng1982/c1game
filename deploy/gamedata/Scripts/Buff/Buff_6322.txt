local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and not IsOnServer then
		local iHp = FilterValue.m_fFoo1;
		if IsSource and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_SHOOT then
			local buffLevel = LuaDataAgt:GetBuffLevel();
			FilterValue.m_fFoo1 = iHp * (1 + buffLevel * 0.04);
		end
	end
	
	return FilterValue;
end
