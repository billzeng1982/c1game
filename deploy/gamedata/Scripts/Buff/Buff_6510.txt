local ThisScriptId = 6510;

local function runBorn(LuaDataAgt)
	local rsource = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local SpeedUP = rsource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local SpeedRatio = 1 + SpeedUP;
	
	LuaDataAgt:ChgSpeedByRatio(SpeedRatio);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_JiaSu", LuaDataAgt, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end
end
