local ThisScriptId = 53;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();

	--buff时间设置
	local itime =source:GetActiveSkillAdditionValue(4);
	if source:IsHaveActiveSkillCheats(1) then     --秘籍1：增加时长
	    itime = itime * 1.1;
	end
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, itime);			--BUFF时间

	--属性加减
	local WitAtk = source:GetWitAtk(VALUE_TYPE.CURRENT);
	local Base1 = source:GetActiveSkillBaseValue(1);
	local Ratio1 = source:GetActiveSkillRateValue(1);
	local Base2 = source:GetActiveSkillBaseValue(2);
	local Ratio2 = source:GetActiveSkillRateValue(2);
	local AddAtk = Base1 + ((WitAtk) * Ratio1);
	local AddDef = Base2 + ((WitAtk) * Ratio2);			--加防的参数

	if LuaDataAgt:Equal(source) and source:IsHaveActiveSkillCheats(2) then     --秘籍2：自身额外加攻防
		AddAtk = AddAtk * 1.5;
		AddDef = AddDef * 1.5;
	end

	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, AddAtk);   --加攻
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddDef);	 --加防



	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();

	--属性加减
	local WitAtk = source:GetWitAtk(VALUE_TYPE.CURRENT);
	local Base1 = source:GetActiveSkillBaseValue(1);
	local Ratio1 = source:GetActiveSkillRateValue(1);
	local Base2 = source:GetActiveSkillBaseValue(2);
	local Ratio2 = source:GetActiveSkillRateValue(2);
	local AddAtk = Base1 + ((WitAtk) * Ratio1);
	local AddDef = Base2 + ((WitAtk) * Ratio2);			--加防的参数

	if LuaDataAgt:Equal(source) and source:IsHaveActiveSkillCheats(2) then     --秘籍2：自身额外加攻防
		AddAtk = AddAtk * 1.5;
		AddDef = AddDef * 1.5;
	end

	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddAtk);   --加攻
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddDef);	 --加防

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
