--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
			
	if iFilterType == FILTER_TYPE.HP and FilterValue.m_iValueChgType == VALUE_CHG_TYPE.CHG_HP_ATK_NORMAL then
		if IsSource then --判断伤害是不是自己造成的
			local CurrHp = LuaDataAgt:GetHp(VALUE_TYPE.CURRENT);
			
			local source = LuaDataAgt:GetBuffSource();
			local Ratio = source:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
			local AddDMG = CurrHp * Ratio;
			
			FilterValue.m_fFoo1 = FilterValue.m_fFoo1 - AddDMG;  --对伤害进行处理
		end
	end
	
	return FilterValue;
end
