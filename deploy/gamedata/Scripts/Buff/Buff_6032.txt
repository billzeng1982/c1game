local ThisScriptId = 6032;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 999);

	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;

	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, AddStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, AddWitAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_QuanShuXingSS", 3 , TarPos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_QuanShuXingSS", LuaDataAgt, TarPos, TarPos, 3, true);
	end
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();

	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	local AddWitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;

	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -AddStrAtk);
	LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT, -AddWitAtk);
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
