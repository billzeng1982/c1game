local ThisScriptId = 6812;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	
	local DodgeChance = rSource: GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_DODGE, DodgeChance);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		--LuaDataAgt:Play2DFXFollowTroopInfo("Prefabs/BasicEffect/UI_CharEff_WuFangShangSheng", 3 , TarPos);
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuFangShangSheng", LuaDataAgt, TarPos, TarPos, 3, true);
	end
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)

	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	
	local DodgeChance = rSource: GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	LuaDataAgt:ChgAttribute(VALUE_TYPE.CURRENT, ATTR_TYPE.ATTR_CHANCE_DODGE, -DodgeChance);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

	return true;
end

end

Buff_6812 = {

	func_checkDeadCond = checkDeadCond;

	func_takeEffect = takeEffect;
}
