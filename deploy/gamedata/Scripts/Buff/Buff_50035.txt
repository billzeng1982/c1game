local ThisScriptId = 50035;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	
	local WitAtk = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF);
	print("1111111111111111111111-----"..WitAtk);
	local WitAtkChgValue = WitAtk * LuaDataAgt:GetResTacticsBuffValue(ThisScriptId);
	LuaDataAgt:SetShareValue(0, WitAtkChgValue);
	
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)
	
	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
    
	local WitAtkClear =  LuaDataAgt:GetShareValue(0);
    LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT,-WitAtkClear);

end

--Buff生效条件判断
local function checkEffectCond(LuaDataAgt)
	return true;
end

--Buff生效
local function takeEffect(LuaDataAgt)

    local WitAtkAdd =  LuaDataAgt:GetShareValue(0);
    LuaDataAgt:ChgWitAtk(VALUE_TYPE.CURRENT,WitAtkAdd);
	print("2222222222222222222-----"..WitAtkAdd);
	
	local TarPos = LuaDataAgt:GetCurPos();
	LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaGongShangSheng", LuaDataAgt, TarPos, TarPos, 6, true); -- 延迟0播该特效
	print("a a a a a a a a a a a a a a a a"..LuaDataAgt:GetShareValue(0));

end


Buff_50035 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
}
