--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local Ratio = rSource: GetPassiveSkillBaseValue(6820, 1) / 100;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * Ratio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);

	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();

		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangXiaJiang", LuaDataAgt, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end
end
