--[[
名称：冲锋陷阵
类型：武将技能Buff
描述：提升移速和迎击伤害
--]]
local ThisScriptId = 25;

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource(); 
	
	--设置buff时间
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 10);
	
	--加速100%
	LuaDataAgt:ChgSpeedByRatio(2);
	
	--加攻
	local apOrigin = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, apOrigin);
	
	--设置Buff过滤
	LuaDataAgt:SetFilter(FILTER_TYPE.POS,TARGET_GROUP.NONE, 0);
	
	-- 设置Buff公式
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)
	
	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	--属性加减回退
	
	--减速100%
	LuaDataAgt:ChgSpeedByRatio(-2);
	
	--减攻
	local apOrigin = LuaDataAgt:GetWitAtk(VALUE_TYPE.ORIGIN_CONF);
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, -apOrigin);
	
	--部队死亡
	LuaDataAgt:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_BUFF, ThisScriptId, 0);
end

--Buff生效条件判断
local function checkEffectCond(LuaDataAgt)
	return true;
end

--Buff生效
local function takeEffect(LuaDataAgt)
	
end

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.POS then
		local iPosZ = FilterValue.m_fFoo3;
		if LuaDataAgt:GetGoGroup() == PLAYER_GROUP.EnumToInt(PLAYER_GROUP.PLAYER_GROUP_DOWN) then
			if iPosZ <= 10 then
				FilterValue.m_fFoo3 = 10;
			end
		else
			if iPosZ >= 70 then
				FilterValue.m_fFoo3 = 70;
			end
		end
	end
	
	return FilterValue;
end

--Buff公式处理
local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	local iRet = 0;
	
	local curHp = rTarget:GetHp(VALUE_TYPE.CURRENT);
	iRet = -curHp;
	
	return iRet;
end

Buff_25 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
	
	func_doFilter = doFilter;
	func_doFormula = doFormula;
}
