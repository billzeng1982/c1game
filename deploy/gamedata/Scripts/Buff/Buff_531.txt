--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	
	if iFilterType == FILTER_TYPE.HP and not IsSource then
		local iHp = FilterValue.m_fFoo1;
		if iHp < 0 then 
		    local source = LuaDataAgt:GetBuffSource();
		    local extra = source:GetActiveSkillCheatsValue(1);
			iHp = iHp * (1 + extra / 100);
			FilterValue.m_fFoo1 = iHp;
		end
	end
	
	return FilterValue;
end
