 -- Buff生效
 local function takeEffect(LuaDataAgt)
	LuaDataAgt:ClearTarget();
	local targetList = LuaDataAgt:FindTargetEx(TARGET_GROUP.ENEMY, SHAPE_TYPE.CIRCLE, 0, 0, 20, 0);  --- 15 = （20 + 部队大小）/ 2
	--剔除死亡或回城部队，targetList也会随之改变
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
	
	local source = LuaDataAgt:GetBuffSource();
	local dancePower = source:GetActiveSkillShareValue(0);
	
	for i=0, targetList.Count-1 do
		targetList:get_Item(i):ChgHp(VALUE_TYPE.CURRENT, dancePower, source, VALUE_CHG_TYPE.CHG_HP_GENERALSKILL, 9, 0); --使用黄月英主动技里面的公式
	end
 end
