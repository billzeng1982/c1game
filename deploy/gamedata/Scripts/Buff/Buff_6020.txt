local ThisScriptId = 6020; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local source = LuaDataAgt:GetBuffSource();
	local iTime = 10;
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.CURRENT,10);
	--属性加减
	local AddStrAtkRatio = LuaDataAgt:GetPassiveSkillBaseValue(ThisScriptId,1) / 100;
	local AddStrAtk = LuaDataAgt:GetStrAtk(VALUE_TYPE.ORIGIN_CONF) * AddStrAtkRatio;
	LuaDataAgt:ChgStrAtk(VALUE_TYPE.CURRENT, AddStrAtk);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_WuGongShangSheng", LuaDataAgt, TarPos, TarPos, 0, true); -- 延迟0播该特效
	end
end
