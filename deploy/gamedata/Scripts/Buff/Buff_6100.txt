--[[
名称:狂热迷信
类型:被动技能
描述:被于吉大招命中的敌人法抗降低 触发机制写在于吉大招里id10 buff写在buff6100
--]]
local ThisScriptId = 6100; 

--Buff诞生处理
local function runBorn(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
		local TarPos = LuaDataAgt:GetCurPos();
		LuaDataAgt:PlayFX("Prefabs/BasicEffect/3D_CharEff_FaFangXiaJiang", LuaDataAgt, TarPos, TarPos, 1, true); -- 延迟1播该特效	
	end
end

--Buff驱散条件判断
local function checkDeadCond(LuaDataAgt)
	
	return false;
end

--Buff驱散生效
local function runDead(LuaDataAgt)
	local rSource = LuaDataAgt:GetBuffSource();
	
	--属性加减
	local PassiveSkillRatio = rSource:GetPassiveSkillBaseValue(ThisScriptId, 1) / 100;
	
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * PassiveSkillRatio;
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, AddWitDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end

--Buff生效条件判断
local function checkEffectCond(LuaDataAgt)
	return true;
end

--Buff生效
local function takeEffect(LuaDataAgt)
	
end

Buff_6100 = {
	func_runBorn = runBorn;
	func_checkDeadCond = checkDeadCond;
	func_runDead = runDead;
	func_checkEffectCond = checkEffectCond;
	func_takeEffect = takeEffect;
}
