local function runBorn(LuaDataAgt)

	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 1);
	
	local speedValue = 1 - source:GetActiveSkillCheatsValue(1)/100;
	LuaDataAgt:ChgSpeedByRatio(speedValue);
end
