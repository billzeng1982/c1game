local function takeEffect(LuaDataAgt)
	if IsOnServer then
		LuaDataAgt:SetSkillEnable(false);
		return;
	end
	--一场战斗中只生效一次
	LuaDataAgt:SetSkillEnable(false);

end

--Buff过滤处理
local function doFilter(LuaDataAgt, iFilterType, rFilterOwner, FilterValue)
	-- Filter所有者是否是发起对象
	local IsSource = rFilterOwner:Equal(FilterValue:GetSource());
	if iFilterType == FILTER_TYPE.HP then
		if not IsOnServer and not IsSource  then
			local iHp = FilterValue.m_fFoo1;
			if iHp < 0 then
				local Ratio = LuaDataAgt:GetActiveSkillBaseValue(1) / 100;
				local iHpback= iHp * Ratio;
				local Source = FilterValue:GetSource();
				local MaxDamge = LuaDataAgt:GetStrAtk(VALUE_TYPE.CURRENT);
				MaxDamge = 5 * MaxDamge; -- 最高自身物攻5倍伤害
				--local SourceHp = Source: GetHp(VALUE_TYPE.RUNTIME_CONF)*0.2;
				--print("SourceHp="..SourceHp);
				iHpback = math.min(-iHpback,MaxDamge); --保护，以免反伤值过大
				if not IsOnServer then
				Source:ChgHp(VALUE_TYPE.CURRENT, -iHpback, Source, VALUE_CHG_TYPE.CHG_HP_COMM, 0, 0);
				end
			end
		end
	end
	return FilterValue;
end
