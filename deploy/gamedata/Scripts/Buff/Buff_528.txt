local function runBorn(LuaDataAgt)
    local source = LuaDataAgt:GetBuffSource();
	
	local cheatsValue = source:GetActiveSkillCheatsValue(2);
	
	if not IsOnServer then
		--增加复活时间
		LuaDataAgt:ChgRevivalTime(VALUE_TYPE.ORIGIN_CONF, cheatsValue);
	end
end
