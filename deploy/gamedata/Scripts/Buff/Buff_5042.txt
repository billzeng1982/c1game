local ThisScriptId = 5042;

 --Buff诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetBuffLifeTime(VALUE_TYPE.RUNTIME_CONF, 5);

	local AddStrDef = LuaDataAgt:GetStrDef(VALUE_TYPE.ORIGIN_CONF) * 0.15;
	local AddWitDef = LuaDataAgt:GetWitDef(VALUE_TYPE.ORIGIN_CONF) * 0.15;
	LuaDataAgt:ChgStrDef(VALUE_TYPE.CURRENT, -AddStrDef);
	LuaDataAgt:ChgWitDef(VALUE_TYPE.CURRENT, -AddWitDef);
	
	if not IsOnServer then
		LuaDataAgt:RefreshFightNum();
	end
end
