local function runBorn(LuaDataAgt)
	
	local maxHp = LuaDataAgt:GetHp(VALUE_TYPE.RUNTIME_CONF);
	print("1111111111111111111111-----"..maxHp);
	local maxHpChgValue = maxHp * LuaDataAgt:GetResTacticsBuffValue(ThisScriptId);
	LuaDataAgt:SetShareValue(0, maxHpChgValue);
    --加血量比率

end
