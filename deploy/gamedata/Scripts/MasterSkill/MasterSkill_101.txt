名称：援军之阵

描述：立即回复我方部队百分比血量

local ThisScriptId = 101;

--技能诞生处理
local function runBorn(LuaDataAgt)
	LuaDataAgt:SetFormula(VALUE_CHG_TYPE.CHG_HP_MASTERSKILL, ThisScriptId, 0);
end

--技能条件判断
local function checkCond(LuaDataAgt)
	return true;
end

--找对象，上面条件满足后执行
local function findTarget(LuaDataAgt)
	LuaDataAgt:ClearTarget();
    local targetList = LuaDataAgt:FindTarget(TARGET_GROUP.SELF);
	--去掉死亡城内部队
	LuaDataAgt:RemoveTroopDeadOrRetreat(targetList);
end

--技能生效
local function takeEffect(LuaDataAgt)
	local targetList = LuaDataAgt:GetTargetList(TARGET_GROUP.SELF);
	local count = targetList.Count; 
	if count <= 0 then
		return;
	end
	
	for i = 0,count - 1 do
	    local target = targetList:get_Item(i);		
		target:ChgHp(VALUE_TYPE.CURRENT, 0, LuaDataAgt, VALUE_CHG_TYPE.CHG_HP_MASTERSKILL, ThisScriptId);
		target:AddBuff(1010, LuaDataAgt, nil);
	end
end

--技能生效完成
local function takeEffectDone(LuaDataAgt)
	
end

local function doFormula(LuaDataAgt, iFormulaType, iFormulaPara, rSource, rTarget, iDamageRef)
	iRet = 3000;
	return iRet;
end

MasterSkill_101 = {
	func_runBorn = runBorn;
	func_checkCond = checkCond;
	func_findTarget = findTarget;
	func_takeEffect = takeEffect;
	func_takeEffectDone = takeEffectDone;
	func_doFormula = doFormula;
}
